24#[derive(Debug, Clone, PartialEq, Eq, Hash)]
25struct BatchKey {
26    service: String,
27    minute_bucket: i64,
28}
29
30impl BatchKey {
31    fn from_metadata<M: BatchMetadata>(metadata: &M) -> Self {
32        let bucket = if metadata.first_timestamp_nanos() > 0 {
33            metadata.first_timestamp_nanos() / 60_000_000_000
34        } else {
35            0
36        };
37
38        Self {
39            service: metadata.service_name().as_ref().to_string(),
40            minute_bucket: bucket,
41        }
42    }
43}

104/// Signal-specific logic used by the batching layer.
105pub trait SignalProcessor {
106    type Request;
107    type Metadata: BatchMetadata;
108
109    fn estimate_row_count(request: &Self::Request) -> usize;
110    fn convert_request(
111        request: &Self::Request,
112        capacity_hint: usize,
113    ) -> Result<(Vec<RecordBatch>, Self::Metadata)>;
114}
115
116type BatchIngestResult<M> = Result<(Vec<CompletedBatch<M>>, M)>;
117
118/// Log-specific signal processor.
119#[derive(Debug, Clone, Copy, Default)]
120pub struct LogSignalProcessor;
121
122impl SignalProcessor for LogSignalProcessor {
123    type Request = ExportLogsServiceRequest;
124    type Metadata = LogMetadata;
125
126    fn estimate_row_count(request: &Self::Request) -> usize {
127        estimate_request_row_count(request)
128    }
129
130    fn convert_request(
131        request: &Self::Request,
132        capacity_hint: usize,
133    ) -> Result<(Vec<RecordBatch>, Self::Metadata)> {
134        let (batch, metadata) = convert_request_to_arrow_with_capacity(request, capacity_hint)
135            .context("Failed to convert OTLP request to Arrow")?;
136        Ok((vec![batch], metadata))
137    }
138}

140/// Trace-specific signal processor placeholder.
141#[derive(Debug, Clone, Copy, Default)]
142pub struct TraceSignalProcessor;
143
144impl SignalProcessor for TraceSignalProcessor {
145    type Request = TraceRequest;
146    type Metadata = TraceMetadata;
147
148    fn estimate_row_count(request: &Self::Request) -> usize {
149        request
150            .resource_spans
151            .iter()
152            .map(|resource_spans| {
153                resource_spans
154                    .scope_spans
155                    .iter()
156                    .map(|scope_spans| scope_spans.spans.len())
157                    .sum::<usize>()
158            })
159            .sum()
160    }
161
162    fn convert_request(
163        request: &Self::Request,
164        _capacity_hint: usize,
165    ) -> Result<(Vec<RecordBatch>, Self::Metadata)> {
166        let (batches, metadata) = TraceArrowConverter::convert(request)?;
167        Ok((batches, metadata))
168    }
169}

181/// Thread-safe batch orchestrator shared across handlers.
182pub struct BatchManager<P: SignalProcessor = LogSignalProcessor> {
183    config: BatchConfig,
184    inner: Arc<Mutex<HashMap<BatchKey, BufferedBatch<P::Metadata>>>>,
185    _marker: PhantomData<P>,
186}
187
188impl<P: SignalProcessor> BatchManager<P> {
189    pub fn new(config: BatchConfig) -> Self {
190        Self {
191            config,
192            inner: Arc::new(Mutex::new(HashMap::new())),
193            _marker: PhantomData,
194        }
195    }
196
197    pub fn ingest(
198        &self,
199        request: &P::Request,
200        approx_bytes: usize,
201    ) -> BatchIngestResult<P::Metadata> {
202        let capacity_hint = P::estimate_row_count(request);
203        let (batches, metadata) = P::convert_request(request, capacity_hint)?;
204
205        if metadata.record_count() == 0 {
206            return Ok((Vec::new(), metadata));
207        }
208
209        let key = BatchKey::from_metadata(&metadata);
210        let mut guard = self.inner.lock();
211        let buffered = guard
212            .entry(key.clone())
213            .or_insert_with(|| BufferedBatch::new(&metadata));
214        buffered.add_batches(batches, &metadata, approx_bytes);
215
216        let mut completed = Vec::new();
217        if buffered.should_flush(&self.config) {
218            let batch = guard
219                .remove(&key)
220                .ok_or_else(|| anyhow!("batch evicted before flush: {:?}", key))?;
221            completed.push(batch.finalize()?);
222        }
223
224        drop(guard);
225
226        Ok((completed, metadata))
227    }
228
229    pub fn drain_expired(&self) -> Result<Vec<CompletedBatch<P::Metadata>>> {
230        let mut guard = self.inner.lock();
231        let mut completed = Vec::new();
232        let keys: Vec<BatchKey> = guard
233            .iter()
234            .filter(|(_, batch)| batch.should_flush(&self.config))
235            .map(|(key, _)| key.clone())
236            .collect();
237
238        for key in keys {
239            if let Some(batch) = guard.remove(&key) {
240                completed.push(batch.finalize()?);
241            }
242        }
243
244        Ok(completed)
245    }
246}

248/// Lightweight helper when batching is disabled entirely.
249///
250/// Converts each OTLP request directly to an Arrow batch without accumulation.
251#[derive(Debug, Clone)]
252pub struct PassthroughBatcher<P: SignalProcessor = LogSignalProcessor>(PhantomData<P>);
253
254impl<P: SignalProcessor> Default for PassthroughBatcher<P> {
255    fn default() -> Self {
256        Self(PhantomData)
257    }
258}
259
260impl<P: SignalProcessor> PassthroughBatcher<P> {
261    pub fn ingest(&self, request: &P::Request) -> Result<CompletedBatch<P::Metadata>> {
262        let capacity_hint = P::estimate_row_count(request);
263        let (batches, metadata) = P::convert_request(request, capacity_hint)?;
264
265        Ok(CompletedBatch { batches, metadata })
266    }
267}

323    #[test]
324    fn test_batch_manager_accumulation() {
325        let config = BatchConfig {
326            max_rows: 100,
327            max_bytes: 1024 * 1024,
328            max_age: Duration::from_secs(10),
329        };
330        let manager = BatchManager::<LogSignalProcessor>::new(config);
331
332        // First request - should not flush
333        let request1 = create_test_request("test-service", 10);
334        let approx1 = request1.encoded_len();
335        let (completed1, _meta1) = manager.ingest(&request1, approx1).unwrap();
336        assert_eq!(completed1.len(), 0); // Not flushed yet
337
338        // Second request - should not flush (total 20 rows)
339        let request2 = create_test_request("test-service", 10);
340        let approx2 = request2.encoded_len();
341        let (completed2, _meta2) = manager.ingest(&request2, approx2).unwrap();
342        assert_eq!(completed2.len(), 0); // Still not flushed
343
344        // Third test with smaller limit - should flush when hitting threshold
345        let config_small = BatchConfig {
346            max_rows: 20,
347            max_bytes: 1024 * 1024,
348            max_age: Duration::from_secs(10),
349        };
350        let manager_small = BatchManager::<LogSignalProcessor>::new(config_small);
351
352        let req1 = create_test_request("test-service", 10);
353        let approx_small_1 = req1.encoded_len();
354        let (c1, _) = manager_small.ingest(&req1, approx_small_1).unwrap();
355        assert_eq!(c1.len(), 0); // 10 rows < 20, no flush
356
357        let req2 = create_test_request("test-service", 10);
358        let approx_small_2 = req2.encoded_len();
359        let (c2, _) = manager_small.ingest(&req2, approx_small_2).unwrap();
360        assert_eq!(c2.len(), 1); // 10 + 10 = 20 rows, should flush!
361        assert_eq!(
362            c2[0].batches.iter().map(|b| b.num_rows()).sum::<usize>(),
363            20
364        );
365    }
