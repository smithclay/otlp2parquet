624/// AggregationTemporality defines how a metric aggregator reports aggregated
625/// values. It describes how those values relate to the time interval over
626/// which they are aggregated.
627#[derive(serde::Serialize, serde::Deserialize)]
628#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, PartialOrd, Ord, ::prost::Enumeration)]
629#[repr(i32)]
630pub enum AggregationTemporality {
631    /// UNSPECIFIED is the default AggregationTemporality, it MUST not be used.
632    Unspecified = 0,
633    /// DELTA is an AggregationTemporality for a metric aggregator which reports
634    /// changes since last report time. Successive metrics contain aggregation of
635    /// values from continuous and non-overlapping intervals.
636    ///
637    /// The values for a DELTA metric are based only on the time interval
638    /// associated with one measurement cycle. There is no dependency on
639    /// previous measurements like is the case for CUMULATIVE metrics.
640    ///
641    /// For example, consider a system measuring the number of requests that
642    /// it receives and reports the sum of these requests every second as a
643    /// DELTA metric:
644    ///
645    ///    1. The system starts receiving at time=t_0.
646    ///    2. A request is received, the system measures 1 request.
647    ///    3. A request is received, the system measures 1 request.
648    ///    4. A request is received, the system measures 1 request.
649    ///    5. The 1 second collection cycle ends. A metric is exported for the
650    ///       number of requests received over the interval of time t_0 to
651    ///       t_0+1 with a value of 3.
652    ///    6. A request is received, the system measures 1 request.
653    ///    7. A request is received, the system measures 1 request.
654    ///    8. The 1 second collection cycle ends. A metric is exported for the
655    ///       number of requests received over the interval of time t_0+1 to
656    ///       t_0+2 with a value of 2.
657    Delta = 1,
658    /// CUMULATIVE is an AggregationTemporality for a metric aggregator which
659    /// reports changes since a fixed start time. This means that current values
660    /// of a CUMULATIVE metric depend on all previous measurements since the
661    /// start time. Because of this, the sender is required to retain this state
662    /// in some form. If this state is lost or invalidated, the CUMULATIVE metric
663    /// values MUST be reset and a new fixed start time following the last
664    /// reported measurement time sent MUST be used.
665    ///
666    /// For example, consider a system measuring the number of requests that
667    /// it receives and reports the sum of these requests every second as a
668    /// CUMULATIVE metric:
669    ///
670    ///    1. The system starts receiving at time=t_0.
671    ///    2. A request is received, the system measures 1 request.
672    ///    3. A request is received, the system measures 1 request.
673    ///    4. A request is received, the system measures 1 request.
674    ///    5. The 1 second collection cycle ends. A metric is exported for the
675    ///       number of requests received over the interval of time t_0 to
676    ///       t_0+1 with a value of 3.
677    ///    6. A request is received, the system measures 1 request.
678    ///    7. A request is received, the system measures 1 request.
679    ///    8. The 1 second collection cycle ends. A metric is exported for the
680    ///       number of requests received over the interval of time t_0 to
681    ///       t_0+2 with a value of 5.
682    ///    9. The system experiences a fault and loses state.
683    ///    10. The system recovers and resumes receiving at time=t_1.
684    ///    11. A request is received, the system measures 1 request.
685    ///    12. The 1 second collection cycle ends. A metric is exported for the
686    ///       number of requests received over the interval of time t_1 to
687    ///       t_0+1 with a value of 1.
688    ///
689    /// Note: Even though, when reporting changes since last report time, using
690    /// CUMULATIVE is valid, it is not recommended. This may cause problems for
691    /// systems that do not use start_time to determine when the aggregation
692    /// value was reset (e.g. Prometheus).
693    Cumulative = 2,
694}
695impl AggregationTemporality {
696    /// String value of the enum field names used in the ProtoBuf definition.
697    ///
698    /// The values are not transformed in any way and thus are considered stable
699    /// (if the ProtoBuf definition does not change) and safe for programmatic use.
700    pub fn as_str_name(&self) -> &'static str {
701        match self {
702            Self::Unspecified => "AGGREGATION_TEMPORALITY_UNSPECIFIED",
703            Self::Delta => "AGGREGATION_TEMPORALITY_DELTA",
704            Self::Cumulative => "AGGREGATION_TEMPORALITY_CUMULATIVE",
705        }
706    }
707    /// Creates an enum from field names used in the ProtoBuf definition.
708    pub fn from_str_name(value: &str) -> ::core::option::Option<Self> {
709        match value {
710            "AGGREGATION_TEMPORALITY_UNSPECIFIED" => Some(Self::Unspecified),
711            "AGGREGATION_TEMPORALITY_DELTA" => Some(Self::Delta),
712            "AGGREGATION_TEMPORALITY_CUMULATIVE" => Some(Self::Cumulative),
713            _ => None,
714        }
715    }
716}
