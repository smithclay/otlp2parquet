1// otlp2parquet-core - Platform-agnostic core logic
2//
3// This crate contains the PURE transformation logic for converting
4// OTLP logs to Arrow RecordBatch. No I/O, no async, no serialization.
5//
6// Philosophy (Fred Brooks): "Separate essence from accident"
7// - Essence: OTLP → Arrow transformation (THIS CRATE)
8// - Accident: Serialization format (Parquet), storage, networking (OTHER CRATES)
9//
10// Serialization (Arrow → Parquet) moved to: otlp2parquet-storage
11// Batching/optimization moved to: otlp2parquet-batch
12

23/// Parse OTLP log data and convert to Arrow RecordBatch
24///
25/// This is the PURE core transformation logic: OTLP bytes → Arrow RecordBatch + metadata.
26/// No I/O, no side effects, deterministic for the same input.
27///
28/// # Arguments
29/// * `otlp_bytes` - Raw OTLP data in the specified format
30/// * `format` - Input format (Protobuf, JSON, or JSONL)
31///
32/// # Returns
33/// * `Ok((RecordBatch, LogMetadata))` - Arrow batch and extracted metadata
34/// * `Err` - If parsing or conversion fails
35///
36/// # Philosophy
37/// "Show me your tables, and I won't usually need your flowcharts; they'll be obvious."
38/// - Fred Brooks
39///
40/// This function preserves information flow: we extract metadata during parsing
41/// and return it alongside the Arrow data. No information is lost.
42///
43/// # Example
44/// ```ignore
45/// use otlp2parquet_core::{parse_otlp_to_arrow, InputFormat};
46///
47/// // Protobuf input (most common)
48/// let protobuf_bytes = /* ... */;
49/// let (batch, metadata) = parse_otlp_to_arrow(protobuf_bytes, InputFormat::Protobuf)?;
50///
51/// // JSON input
52/// let json_bytes = br#"{"resourceLogs":[...]}"#;
53/// let (batch, metadata) = parse_otlp_to_arrow(json_bytes, InputFormat::Json)?;
54/// ```
55pub fn parse_otlp_to_arrow(
56    otlp_bytes: &[u8],
57    format: InputFormat,
58) -> Result<(RecordBatch, LogMetadata)> {
59    // Parse the input format into an ExportLogsServiceRequest
60    let request = otlp::parse_otlp_request(otlp_bytes, format)?;
61    convert_request_to_arrow(&request)
62}
63
64/// Convert a parsed OTLP request directly into Arrow structures.
65///
66/// Consumers that already decoded the OTLP payload (e.g. batching code) can call
67/// this helper to avoid the encode/decode round-trip required by
68/// `parse_otlp_to_arrow`.
69pub fn convert_request_to_arrow(
70    request: &otlp2parquet_proto::opentelemetry::proto::collector::logs::v1::ExportLogsServiceRequest,
71) -> Result<(RecordBatch, LogMetadata)> {
72    let capacity_hint = estimate_request_row_count(request);
73    convert_request_to_arrow_with_capacity(request, capacity_hint)
74}
75
76/// Variant of `convert_request_to_arrow` that accepts an explicit capacity hint for pre-allocation.
77///
78/// Callers that already know approximately how many log records will be produced can use this to
79/// avoid recomputing the count inside the converter.
80pub fn convert_request_to_arrow_with_capacity(
81    request: &otlp2parquet_proto::opentelemetry::proto::collector::logs::v1::ExportLogsServiceRequest,
82    capacity_hint: usize,
83) -> Result<(RecordBatch, LogMetadata)> {
84    let mut converter = otlp::ArrowConverter::with_capacity(capacity_hint.max(1));
85    converter.add_from_request(request)?;
86    converter.finish()
87}
88
89/// Estimate the number of log records contained in an OTLP request.
90///
91/// This is used to size Arrow builders before conversion. The count is exact because the OTLP
92/// schema exposes log record vectors directly.
93pub fn estimate_request_row_count(
94    request: &otlp2parquet_proto::opentelemetry::proto::collector::logs::v1::ExportLogsServiceRequest,
95) -> usize {
96    request
97        .resource_logs
98        .iter()
99        .map(|resource_logs| {
100            resource_logs
101                .scope_logs
102                .iter()
103                .map(|scope_logs| scope_logs.log_records.len())
104                .sum::<usize>()
105        })
106        .sum()
107}

109#[cfg(test)]
110mod tests {
111    use super::*;
112
113    #[test]
114    fn test_parse_empty_logs() {
115        // Create minimal valid OTLP request with no logs
116        use otlp2parquet_proto::opentelemetry::proto::collector::logs::v1::ExportLogsServiceRequest;
117        use prost::Message;
118
119        let request = ExportLogsServiceRequest {
120            resource_logs: vec![],
121        };
122
123        let mut bytes = Vec::new();
124        request.encode(&mut bytes).unwrap();
125
126        // Should successfully parse to Arrow
127        let result = parse_otlp_to_arrow(&bytes, InputFormat::Protobuf);
128        if let Err(ref e) = result {
129            eprintln!("Error: {:?}", e);
130        }
131        assert!(result.is_ok());
132
133        let (batch, metadata) = result.unwrap();
134        assert_eq!(batch.num_rows(), 0);
135        assert_eq!(metadata.service_name.as_ref(), "");
136        assert_eq!(metadata.record_count, 0);
137    }
