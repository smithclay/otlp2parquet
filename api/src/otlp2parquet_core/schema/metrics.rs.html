1// ClickHouse-compatible Arrow schemas for OpenTelemetry metrics
2//
3// This module provides separate schemas for each metric type following the
4// reference implementation from duckdb-otlp. Each metric type (gauge, sum,
5// histogram, exponential_histogram, summary) has its own schema that shares
6// a common set of base columns.
7//
8// Reference: https://github.com/smithclay/duckdb-otlp/blob/main/src/schema/otlp_metrics_schemas.hpp
9
10use arrow::datatypes::{DataType, Field, Fields, Schema, TimeUnit};
11use std::collections::HashMap;
12use std::sync::{Arc, OnceLock};
13
14use crate::otlp::field_names::arrow as field;
15
16/// Returns the base fields shared by all metric types
17fn base_fields() -> Vec<Field> {
18    let map_type = map_type();
19
20    vec![
21        // Timestamp - nanosecond precision, UTC
22        Field::new(
23            field::TIMESTAMP,
24            DataType::Timestamp(TimeUnit::Nanosecond, Some("UTC".into())),
25            false,
26        ),
27        // Service identification
28        Field::new(field::SERVICE_NAME, DataType::Utf8, false),
29        // Metric metadata
30        Field::new(field::METRIC_NAME, DataType::Utf8, false),
31        Field::new(field::METRIC_DESCRIPTION, DataType::Utf8, true),
32        Field::new(field::METRIC_UNIT, DataType::Utf8, true),
33        // Resource and scope information
34        Field::new(field::RESOURCE_ATTRIBUTES, map_type.clone(), false),
35        Field::new(field::SCOPE_NAME, DataType::Utf8, true),
36        Field::new(field::SCOPE_VERSION, DataType::Utf8, true),
37        // Data point attributes
38        Field::new(field::ATTRIBUTES, map_type, false),
39    ]
40}
41
42/// Helper function to create a Map<String, String> type
43fn map_type() -> DataType {
44    let entry_fields: Fields = vec![
45        Field::new(field::KEY, DataType::Utf8, false),
46        Field::new(field::VALUE, DataType::Utf8, true),
47    ]
48    .into();
49
50    DataType::Map(
51        Arc::new(Field::new(
52            field::ENTRIES,
53            DataType::Struct(entry_fields),
54            false,
55        )),
56        false,
57    )
58}

60/// Returns the Arrow schema for gauge metrics
61pub fn otel_metrics_gauge_schema() -> Schema {
62    otel_metrics_gauge_schema_arc().as_ref().clone()
63}
64
65/// Returns a cached `Arc<Schema>` for gauge metrics
66pub fn otel_metrics_gauge_schema_arc() -> Arc<Schema> {
67    static SCHEMA: OnceLock<Arc<Schema>> = OnceLock::new();
68    Arc::clone(SCHEMA.get_or_init(|| Arc::new(build_gauge_schema())))
69}
70
71fn build_gauge_schema() -> Schema {
72    let mut fields = base_fields();
73    fields.push(Field::new(field::VALUE_COL, DataType::Float64, false));
74
75    let mut metadata = HashMap::new();
76    metadata.insert(
77        "otlp2parquet.metrics_schema_version".to_string(),
78        "1.0.0".to_string(),
79    );
80    metadata.insert("otlp2parquet.metric_type".to_string(), "gauge".to_string());
81
82    Schema::new_with_metadata(fields, metadata)
83}

97    let mut fields = base_fields();
98    fields.push(Field::new(field::VALUE_COL, DataType::Float64, false));
99    fields.push(Field::new(
100        field::AGGREGATION_TEMPORALITY,
101        DataType::Int32,
102        false,
103    ));
104    fields.push(Field::new(field::IS_MONOTONIC, DataType::Boolean, false));
105
106    let mut metadata = HashMap::new();
107    metadata.insert(
108        "otlp2parquet.metrics_schema_version".to_string(),
109        "1.0.0".to_string(),
110    );
111    metadata.insert("otlp2parquet.metric_type".to_string(), "sum".to_string());
112
113    Schema::new_with_metadata(fields, metadata)
114}

139        false,

142    fields.push(Field::new(field::MAX, DataType::Float64, true));
