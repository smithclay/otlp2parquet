1// Arrow builder construction helpers
2//
3// Provides utilities for creating Arrow builders with the correct schema
4
5use arrow::array::{
6    builder::MapFieldNames, ArrayBuilder, BinaryBuilder, BooleanBuilder, Float64Builder,
7    Int64Builder, LargeStringBuilder, StringBuilder, StructBuilder,
8};
9
10/// Size of OpenTelemetry TraceId in bytes (128 bits)
11pub(crate) const TRACE_ID_SIZE: i32 = 16;
12
13/// Size of OpenTelemetry SpanId in bytes (64 bits)
14pub(crate) const SPAN_ID_SIZE: i32 = 8;
15
16/// Field names for Map types in Arrow schema
17pub(crate) fn map_field_names() -> MapFieldNames {
18    MapFieldNames {
19        entry: "entries".to_string(),
20        key: "key".to_string(),
21        value: "value".to_string(),
22    }
23}
24
25/// Create a new StructBuilder for AnyValue with correct field order and types
26pub(crate) fn new_any_value_struct_builder() -> StructBuilder {
27    let fields = crate::schema::any_value_fields();
28    StructBuilder::new(
29        fields.clone(),
30        vec![
31            Box::new(StringBuilder::new()) as Box<dyn ArrayBuilder>,
32            Box::new(StringBuilder::new()),
33            Box::new(BooleanBuilder::new()),
34            Box::new(Int64Builder::new()),
35            Box::new(Float64Builder::new()),
36            Box::new(BinaryBuilder::new()),
37            Box::new(LargeStringBuilder::new()),
38        ],
39    )
40}
