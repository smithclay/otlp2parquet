1//! Protocol field number constants for OTLP protobuf messages.
2//!
3//! These constants define the field numbers used in OpenTelemetry Protocol (OTLP)
4//! protobuf messages. They are primarily used for validation, error messages, and
5//! low-level protobuf parsing.
6//!
7//! Reference: <https://github.com/open-telemetry/opentelemetry-proto>
8//! Inspired by: <https://github.com/open-telemetry/otel-arrow/tree/main/rust>
9
10/// Protobuf wire type constants
11pub mod wire_types {
12    /// Variable-length integer (int32, int64, uint32, uint64, sint32, sint64, bool, enum)
13    pub const VARINT: u64 = 0;
14    /// 64-bit fixed-length (fixed64, sfixed64, double)
15    pub const FIXED64: u64 = 1;
16    /// Length-delimited (string, bytes, embedded messages, packed repeated fields)
17    pub const LEN: u64 = 2;
18    /// 32-bit fixed-length (fixed32, sfixed32, float)
19    pub const FIXED32: u64 = 5;
20}
