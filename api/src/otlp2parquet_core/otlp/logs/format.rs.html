1use anyhow::Result;

4use crate::otlp::common::{

8/// Parse OTLP logs from bytes in the specified format.
9pub fn parse_otlp_request(bytes: &[u8], format: InputFormat) -> Result<ExportLogsServiceRequest> {
10    parse_request(bytes, format, Some(normalise_json_value))
11}
12
13impl OtlpSignalRequest for ExportLogsServiceRequest {
14    const JSONL_EMPTY_ERROR: &'static str = "JSONL input contained no valid log records";
15
16    fn merge(&mut self, mut other: Self) {
17        self.resource_logs.append(&mut other.resource_logs);
18    }
19
20    fn is_empty(&self) -> bool {
21        self.resource_logs.is_empty()
22    }
23}

62        let request =
