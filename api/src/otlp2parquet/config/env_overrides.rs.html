6/// Abstraction over environment-variable lookups so runtimes without `std::env`
7/// can supply their own source of overrides.
8pub trait EnvSource {
9    fn get(&self, key: &str) -> Option<String>;
10
11    /// Get an environment variable WITHOUT the OTLP2PARQUET_ prefix
12    /// Used for AWS standard variables (AWS_ACCESS_KEY_ID, etc.)
13    fn get_raw(&self, key: &str) -> Option<String>;
14}
15
16/// Apply environment-variable overrides (highest priority) to the runtime config.
17pub fn apply_env_overrides<E: EnvSource>(config: &mut RuntimeConfig, env: &E) -> Result<()> {
18    // Batch configuration
19    if let Some(val) = get_env_usize(env, "BATCH_MAX_ROWS")? {
20        config.batch.max_rows = val;
21    }
22
23    // Server configuration (listen addr, log level/format)
24    if let Some(addr) = get_env_string(env, "LISTEN_ADDR")? {
25        ensure_server(config).listen_addr = addr;
26    }
27    if let Some(level) = get_env_string(env, "LOG_LEVEL")? {
28        ensure_server(config).log_level = level;
29    }
30    if let Some(format) = get_env_string(env, "LOG_FORMAT")? {
31        let parsed = match format.to_lowercase().as_str() {
32            "json" => LogFormat::Json,
33            _ => LogFormat::Text,
34        };
35        ensure_server(config).log_format = parsed;
36    }
37
38    if let Some(val) = get_env_usize(env, "BATCH_MAX_BYTES")? {
39        config.batch.max_bytes = val;
40    }
41    if let Some(val) = get_env_u64(env, "BATCH_MAX_AGE_SECS")? {
42        config.batch.max_age_secs = val;
43    }
44    // Support both BATCH_ENABLED (canonical) and BATCHING_ENABLED (legacy)
45    if let Some(val) = get_env_bool(env, "BATCH_ENABLED")? {
46        config.batch.enabled = val;
47    } else if let Some(val) = get_env_bool(env, "BATCHING_ENABLED")? {
48        config.batch.enabled = val;
49    }
50
51    // Request configuration
52    if let Some(val) = get_env_usize(env, "MAX_PAYLOAD_BYTES")? {
53        config.request.max_payload_bytes = val;
54    }
55
56    // Storage backend
57    if let Some(backend) = get_env_string(env, "STORAGE_BACKEND")? {
58        config.storage.backend = backend
59            .parse::<StorageBackend>()
60            .context("Invalid OTLP2PARQUET_STORAGE_BACKEND value")?;
61    }
62    // Filesystem storage
63    if let Some(path) = get_env_string(env, "STORAGE_PATH")? {
64        if config.storage.fs.is_none() {
65            config.storage.fs = Some(FsConfig::default());
66        }
67        if let Some(ref mut fs) = config.storage.fs {
68            fs.path = path;
69        }
70    }
71
72    // S3 storage
73    if let Some(bucket) = get_env_string(env, "S3_BUCKET")? {
74        ensure_s3(config).bucket = bucket;
75    }
76    if let Some(region) = get_env_string(env, "S3_REGION")? {
77        ensure_s3(config).region = region;
78    }
79    if let Some(endpoint) = get_env_string(env, "S3_ENDPOINT")? {
80        ensure_s3(config).endpoint = Some(endpoint);
81    }
82    if let Some(prefix) = get_env_string(env, "S3_PREFIX")? {
83        ensure_s3(config).prefix = normalize_prefix(prefix);
84    }
85    // Also support generic PREFIX for backwards compatibility
86    if let Some(prefix) = get_env_string(env, "PREFIX")? {
87        ensure_s3(config).prefix = normalize_prefix(prefix);
88    }
89
90    // R2 storage
91    if let Some(bucket) = get_env_string(env, "R2_BUCKET")? {
92        ensure_r2(config).bucket = bucket;
93    }
94    if let Some(account_id) = get_env_string(env, "R2_ACCOUNT_ID")? {
95        ensure_r2(config).account_id = account_id;
96    }
97    // AWS standard credentials (without OTLP2PARQUET_ prefix for compatibility)
98    if let Some(access_key_id) = get_raw_env_string(env, "AWS_ACCESS_KEY_ID")? {
99        ensure_r2(config).access_key_id = access_key_id;
100    }
101    if let Some(secret_access_key) = get_raw_env_string(env, "AWS_SECRET_ACCESS_KEY")? {
102        ensure_r2(config).secret_access_key = secret_access_key;
103    }
104    if let Some(endpoint) = get_raw_env_string(env, "AWS_ENDPOINT_URL")? {
105        ensure_r2(config).endpoint = Some(endpoint);
106    }
107    if let Some(prefix) = get_env_string(env, "R2_PREFIX")? {
108        ensure_r2(config).prefix = normalize_prefix(prefix);
109    }
110
111    Ok(())
112}

114fn ensure_s3(config: &mut RuntimeConfig) -> &mut S3Config {
115    config.storage.s3.get_or_insert_with(|| S3Config {
116        bucket: String::new(),
117        region: String::new(),
118        endpoint: None,
119        prefix: None,
120    })
121}
122
123fn ensure_r2(config: &mut RuntimeConfig) -> &mut R2Config {
124    config.storage.r2.get_or_insert_with(|| R2Config {
125        bucket: String::new(),
126        account_id: String::new(),
127        access_key_id: String::new(),
128        secret_access_key: String::new(),
129        endpoint: None,
130        prefix: None,
131    })
132}

134fn ensure_server(config: &mut RuntimeConfig) -> &mut ServerConfig {
135    config.server.get_or_insert_with(ServerConfig::default)
136}

138fn get_env_string<E: EnvSource>(env: &E, key: &str) -> Result<Option<String>> {
139    Ok(env.get(key))

142/// Get a raw environment variable without the OTLP2PARQUET_ prefix
143/// Used for AWS standard variables like AWS_ACCESS_KEY_ID
144fn get_raw_env_string<E: EnvSource>(env: &E, key: &str) -> Result<Option<String>> {
145    Ok(env.get_raw(key))

148fn get_env_usize<E: EnvSource>(env: &E, key: &str) -> Result<Option<usize>> {
149    match get_env_string(env, key)? {
150        Some(val) => {
151            let parsed = val
152                .parse::<usize>()
153                .map_err(|e| anyhow!("Failed to parse {}{}: {}", ENV_PREFIX, key, e))?;
154            Ok(Some(parsed))
155        }
156        None => Ok(None),
157    }
158}

172fn get_env_bool<E: EnvSource>(env: &E, key: &str) -> Result<Option<bool>> {
173    match get_env_string(env, key)? {
174        Some(val) => {
175            let parsed = val.parse::<bool>().map_err(|e| {
176                anyhow!(
177                    "Failed to parse {}{} (expected bool): {}",
178                    ENV_PREFIX,
179                    key,
180                    e
181                )
182            })?;
183            Ok(Some(parsed))
184        }
185        None => Ok(None),
186    }
187}

189fn normalize_prefix(prefix: String) -> Option<String> {
190    if prefix.is_empty() {
191        None
192    } else if prefix.ends_with('/') {
193        Some(prefix)
194    } else {
195        Some(format!("{}/", prefix))
196    }
