1//! Storage operator initialization and management.
2
3use crate::config::{RuntimeConfig, StorageBackend};
4use once_cell::sync::OnceCell;
5
6use super::error::{Result, WriterError};
7
8static OPERATOR: OnceCell<opendal::Operator> = OnceCell::new();
9static STORAGE_PREFIX: OnceCell<Option<String>> = OnceCell::new();
10
11/// Initialize storage operator from RuntimeConfig.
12pub fn initialize_storage(config: &RuntimeConfig) -> Result<()> {
13    if OPERATOR.get().is_some() {
14        return Ok(());
15    }
16
17    let operator = match config.storage.backend {
18        StorageBackend::Fs => {
19            let fs = config.storage.fs.as_ref().ok_or_else(|| {
20                WriterError::invalid_config("fs config required for filesystem backend".to_string())
21            })?;
22
23            let fs_builder = opendal::services::Fs::default().root(&fs.path);
24            opendal::Operator::new(fs_builder)
25                .map_err(|e| {
26                    WriterError::write_failure(format!(
27                        "Failed to create filesystem operator: {}",
28                        e
29                    ))
30                })?
31                .finish()
32        }
33        StorageBackend::S3 => {
34            let s3 = config.storage.s3.as_ref().ok_or_else(|| {
35                WriterError::invalid_config("s3 config required for S3 backend".to_string())
36            })?;
37
38            let _ = STORAGE_PREFIX.set(s3.prefix.clone());
39
40            let mut s3_builder = opendal::services::S3::default()
41                .bucket(&s3.bucket)
42                .region(&s3.region);
43
44            if let Some(endpoint) = &s3.endpoint {
45                s3_builder = s3_builder.endpoint(endpoint);
46            }
47
48            opendal::Operator::new(s3_builder)
49                .map_err(|e| {
50                    WriterError::write_failure(format!("Failed to create S3 operator: {}", e))
51                })?
52                .finish()
53        }
54        StorageBackend::R2 => {
55            let r2 = config.storage.r2.as_ref().ok_or_else(|| {
56                WriterError::invalid_config("r2 config required for R2 backend".to_string())
57            })?;
58
59            let _ = STORAGE_PREFIX.set(r2.prefix.clone());
60
61            let endpoint = r2
62                .endpoint
63                .clone()
64                .unwrap_or_else(|| format!("https://{}.r2.cloudflarestorage.com", r2.account_id));
65
66            let r2_builder = opendal::services::S3::default()
67                .bucket(&r2.bucket)
68                .region("auto")
69                .endpoint(&endpoint)
70                .access_key_id(&r2.access_key_id)
71                .secret_access_key(&r2.secret_access_key);
72
73            opendal::Operator::new(r2_builder)
74                .map_err(|e| {
75                    WriterError::write_failure(format!("Failed to create R2 operator: {}", e))
76                })?
77                .finish()
78        }
79    };
80
81    match OPERATOR.set(operator) {
82        Ok(_) => {
83            tracing::debug!("Storage operator initialized");
84            Ok(())
85        }
86        Err(_) => {
87            tracing::debug!("Storage operator already initialized by another call");
88            Ok(())
89        }
90    }
91}
92
93/// Get the global storage operator.
94pub(crate) fn get_operator() -> Option<&'static opendal::Operator> {
95    OPERATOR.get()
96}
97
98/// Get the configured storage prefix (e.g., "smoke-abc123/").
99pub(crate) fn get_storage_prefix() -> Option<&'static str> {
100    STORAGE_PREFIX
101        .get()
102        .and_then(|opt| opt.as_ref())
103        .map(|s| s.as_str())
104}
