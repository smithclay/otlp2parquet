14/// Request parameters for writing a batch to storage.
15pub struct WriteBatchRequest<'a> {
16    /// Arrow RecordBatch to write
17    pub batch: &'a RecordBatch,
18    /// Type of OTLP signal (logs, traces, metrics)
19    pub signal_type: SignalType,
20    /// Metric type if signal_type is Metrics (gauge, sum, etc.)
21    pub metric_type: Option<&'a str>,
22    /// Service name for logging (not used for partitioning)
23    pub service_name: &'a str,
24    /// Timestamp in microseconds (from OTLP-to-Arrow nanos_to_micros conversion)
25    pub timestamp_micros: i64,
26}

28/// Write a batch as a Parquet file.
29async fn write_plain_parquet(
30    signal_type: SignalType,
31    metric_type: Option<&str>,
32    service_name: &str,
33    timestamp_micros: i64,
34    batch: &RecordBatch,
35) -> Result<String> {
36    let op = super::storage::get_operator().ok_or_else(|| {
37        WriterError::write_failure(
38            "Storage operator not initialized. Call initialize_storage() with RuntimeConfig before writing."
39                .to_string(),
40        )
41    })?;
42
43    let file_path =
44        generate_parquet_path(signal_type, metric_type, service_name, timestamp_micros)?;
45
46    tracing::debug!("Writing plain Parquet to path: {}", file_path);
47
48    let parquet_bytes = to_parquet_bytes(batch).map_err(|e| {
49        WriterError::write_failure(format!("Failed to encode Parquet bytes: {}", e))
50    })?;
51    let bytes_written = parquet_bytes.len();
52
53    op.write(&file_path, parquet_bytes).await.map_err(|e| {
54        WriterError::write_failure(format!(
55            "Failed to write parquet bytes to '{}': {}",
56            file_path, e
57        ))
58    })?;
59
60    let row_count = batch.num_rows();
61    tracing::info!(
62        "âœ“ Wrote {} rows to '{}' (plain Parquet, {} bytes)",
63        row_count,
64        file_path,
65        bytes_written
66    );
67
68    Ok(file_path)
69}

71pub async fn write_batch(req: WriteBatchRequest<'_>) -> Result<String> {
72    let row_count = req.batch.num_rows();
73
74    tracing::debug!(
75        "Writing {} rows (service: {}, signal: {:?}, metric: {:?})",
76        row_count,
77        req.service_name,
78        req.signal_type,
79        req.metric_type
80    );
81
82    write_plain_parquet(
83        req.signal_type,
84        req.metric_type,
85        req.service_name,
86        req.timestamp_micros,
87        req.batch,
88    )
89    .await
90}

92/// Generate a partitioned file path for plain Parquet files.
93fn generate_parquet_path(
94    signal_type: SignalType,
95    metric_type: Option<&str>,
96    service_name: &str,
97    timestamp_micros: i64,
98) -> Result<String> {
99    let (year, month, day, hour) = partition_from_timestamp(timestamp_micros);
100
101    let signal_prefix: Cow<'_, str> = match signal_type {
102        SignalType::Logs => Cow::Borrowed("logs"),
103        SignalType::Traces => Cow::Borrowed("traces"),
104        SignalType::Metrics => {
105            if let Some(mtype) = metric_type {
106                Cow::Owned(format!("metrics/{}", mtype))
107            } else {
108                Cow::Borrowed("metrics")
109            }
110        }
111    };
112
113    let safe_service = sanitize_service_name(service_name);
114    let suffix = Uuid::new_v4().simple();
115
116    let storage_prefix = super::storage::get_storage_prefix().unwrap_or("");
117
118    Ok(format!(
119        "{}{}/{}/year={}/month={:02}/day={:02}/hour={:02}/{}-{}.parquet",
120        storage_prefix,
121        signal_prefix,
122        safe_service,
123        year,
124        month,
125        day,
126        hour,
127        timestamp_micros,
128        suffix
129    ))
130}

132fn sanitize_service_name(service_name: &str) -> Cow<'_, str> {
133    const INVALID: [char; 10] = ['/', '\\', ' ', ':', '*', '?', '"', '<', '>', '|'];
134
135    if service_name.is_empty() {
136        return Cow::Borrowed("unknown-service");
137    }
138
139    if service_name.chars().any(|c| INVALID.contains(&c)) {
140        let sanitized = service_name
141            .chars()
142            .map(|c| if INVALID.contains(&c) { '_' } else { c })
143            .collect::<String>();
144        Cow::Owned(sanitized)
145    } else {
146        Cow::Borrowed(service_name)
147    }
148}

150fn fallback_partition() -> (i32, u8, u8, u8) {
151    let now = OffsetDateTime::now_utc();
152    (now.year(), u8::from(now.month()), now.day(), now.hour())
153}
154
155fn partition_from_timestamp(timestamp_micros: i64) -> (i32, u8, u8, u8) {
156    if timestamp_micros <= 0 {
157        return fallback_partition();
158    }
159
160    let nanos = i128::from(timestamp_micros).saturating_mul(1_000);
161    match OffsetDateTime::from_unix_timestamp_nanos(nanos) {
162        Ok(dt) => (dt.year(), u8::from(dt.month()), dt.day(), dt.hour()),
163        Err(_) => fallback_partition(),
164    }
165}

171    #[test]
172    fn test_extract_timestamp_from_arrow_batch() {
173        use arrow::array::{ArrayRef, TimestampNanosecondArray};
174        use arrow::datatypes::{DataType, Field, Schema};
175        use std::sync::Arc;
176
177        let timestamp_nanos = 1_736_938_800_000_000_000i64;
178        let timestamp_array = TimestampNanosecondArray::from(vec![timestamp_nanos]);
179        let dummy_array = arrow::array::StringArray::from(vec!["test"]);
180
181        let schema = Schema::new(vec![
182            Field::new(
183                "timestamp",
184                DataType::Timestamp(arrow::datatypes::TimeUnit::Nanosecond, None),
185                false,
186            ),
187            Field::new("service", DataType::Utf8, false),
188        ]);
189
190        let batch = RecordBatch::try_new(
191            Arc::new(schema),
192            vec![
193                Arc::new(timestamp_array) as ArrayRef,
194                Arc::new(dummy_array) as ArrayRef,
195            ],
196        )
197        .unwrap();
198
199        if let Some(array) = batch
200            .column(0)
201            .as_any()
202            .downcast_ref::<TimestampNanosecondArray>()
203        {
204            let value = array.value(0);
205            assert_eq!(value, timestamp_nanos);
206            assert_eq!(value.to_string().len(), 19);
207
208            let timestamp_ms = value / 1_000_000;
209            assert_eq!(timestamp_ms, 1_736_938_800_000);
210            assert_eq!(timestamp_ms.to_string().len(), 13);
211        } else {
212            panic!("Failed to extract timestamp from batch");
213        }
214    }
