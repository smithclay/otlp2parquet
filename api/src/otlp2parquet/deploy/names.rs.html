1//! Fun default name generation for stacks and workers
2
3use rand::Rng;
4
5#[allow(dead_code)]
6const ADJECTIVES: &[&str] = &[
7    "swift", "eager", "bright", "cosmic", "dapper", "fluent", "golden", "humble", "jovial", "keen",
8    "lively", "mellow", "nimble", "plucky", "quick", "rustic", "snappy", "trusty", "vivid",
9    "witty",
10];
11
12#[allow(dead_code)]
13const NOUNS: &[&str] = &[
14    "arrow", "beacon", "conduit", "depot", "emitter", "funnel", "gauge", "harbor", "inlet",
15    "journal", "keeper", "ledger", "metric", "nexus", "outlet", "parquet", "queue", "relay",
16    "signal", "tracer",
17];
18
19/// Generate a fun default name like "nimble-relay-2847"
20#[allow(dead_code)]
21pub fn generate() -> String {
22    let mut rng = rand::thread_rng();
23    let adjective = ADJECTIVES[rng.gen_range(0..ADJECTIVES.len())];
24    let noun = NOUNS[rng.gen_range(0..NOUNS.len())];
25    let number: u16 = rng.gen_range(1000..10000);
26    format!("{}-{}-{}", adjective, noun, number)
27}
28
29#[cfg(test)]
30mod tests {
31    use super::*;
32
33    #[test]
34    fn test_generate_name_format() {
35        let name = generate();
36        let parts: Vec<&str> = name.split('-').collect();
37        assert_eq!(parts.len(), 3);
38        assert!(ADJECTIVES.contains(&parts[0]));
39        assert!(NOUNS.contains(&parts[1]));
40        let number: u16 = parts[2].parse().unwrap();
41        assert!((1000..10000).contains(&number));
42    }
43}
