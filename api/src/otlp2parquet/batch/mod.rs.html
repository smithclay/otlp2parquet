1//! In-memory batch accumulation for server mode.
2//!
3//! Accumulates Arrow batches in memory and merges them into larger Arrow batches.
4//! This reduces the number of storage writes and improves compression efficiency.
5//!
6//! Note: This module provides the batching infrastructure for when `batch.enabled=true`
7//! in the server config. Currently the handlers write directly per-request, but this
8//! infrastructure is available for future use.
9

24#[derive(Debug, Clone, PartialEq, Eq, Hash)]
25struct BatchKey {
26    service: String,
27    minute_bucket: i64,
28}
29
30impl BatchKey {
31    fn from_metadata<M: BatchMetadata>(metadata: &M) -> Self {
32        let bucket = if metadata.first_timestamp_micros() > 0 {
33            // Metadata timestamps are stored in microseconds; bucket by minute in micros.
34            metadata.first_timestamp_micros() / 60_000_000
35        } else {
36            0
37        };
38
39        Self {
40            service: metadata.service_name().as_ref().to_string(),
41            minute_bucket: bucket,
42        }
43    }
44}

62/// Metadata required by the batching layer.
63pub trait BatchMetadata: Clone {
64    fn service_name(&self) -> &Arc<str>;
65    /// Stored in microseconds.
66    fn first_timestamp_micros(&self) -> i64;
67    fn record_count(&self) -> usize;
68    fn aggregate(service_name: Arc<str>, first_timestamp_micros: i64, record_count: usize) -> Self;
69}
70
71impl BatchMetadata for LogMetadata {
72    fn service_name(&self) -> &Arc<str> {
73        &self.service_name
74    }
75
76    fn first_timestamp_micros(&self) -> i64 {
77        self.first_timestamp_micros
78    }
79
80    fn record_count(&self) -> usize {
81        self.record_count
82    }
83
84    fn aggregate(service_name: Arc<str>, first_timestamp_micros: i64, record_count: usize) -> Self {
85        Self {
86            service_name,
87            first_timestamp_micros,
88            record_count,
89        }
90    }

93/// Signal-specific logic used by the batching layer.
94pub trait SignalProcessor {
95    type Request;
96    type Metadata: BatchMetadata;
97
98    fn estimate_row_count(request: &Self::Request) -> usize;
99    fn convert_request(
100        request: &Self::Request,
101        capacity_hint: usize,
102    ) -> Result<(Vec<RecordBatch>, Self::Metadata)>;
103}

105type BatchIngestResult<M> = Result<(Vec<CompletedBatch<M>>, M)>;
106
107/// Log-specific signal processor using Arrow-native PartitionedBatch.
108#[derive(Debug, Clone, Copy, Default)]
109pub struct LogSignalProcessor;
110
111impl SignalProcessor for LogSignalProcessor {
112    type Request = PartitionedBatch;
113    type Metadata = LogMetadata;
114
115    fn estimate_row_count(request: &Self::Request) -> usize {
116        request.record_count
117    }
118
119    fn convert_request(
120        request: &Self::Request,
121        _capacity_hint: usize,
122    ) -> Result<(Vec<RecordBatch>, Self::Metadata)> {
123        let metadata = LogMetadata {
124            service_name: Arc::clone(&request.service_name),
125            first_timestamp_micros: request.min_timestamp_micros,
126            record_count: request.record_count,
127        };
128        Ok((vec![request.batch.clone()], metadata))
129    }
130}

142/// Thread-safe batch orchestrator shared across handlers.
143pub struct BatchManager<P: SignalProcessor = LogSignalProcessor> {
144    config: BatchConfig,
145    inner: Arc<Mutex<BatchState<P>>>,
146    _marker: PhantomData<P>,
147}
148
149#[derive(Debug)]
150struct BatchState<P: SignalProcessor> {
151    batches: HashMap<BatchKey, BufferedBatch<P::Metadata>>,
152    total_bytes: usize,
153}
154
155impl<P: SignalProcessor> BatchManager<P> {
156    pub fn new(config: BatchConfig) -> Self {
157        Self {
158            config,
159            inner: Arc::new(Mutex::new(BatchState {
160                batches: HashMap::new(),
161                total_bytes: 0,
162            })),
163            _marker: PhantomData,
164        }
165    }
166
167    pub fn ingest(
168        &self,
169        request: &P::Request,
170        approx_bytes: usize,
171    ) -> BatchIngestResult<P::Metadata> {
172        let capacity_hint = P::estimate_row_count(request);
173        let (batches, metadata) = P::convert_request(request, capacity_hint)?;
174
175        if metadata.record_count() == 0 {
176            return Ok((Vec::new(), metadata));
177        }
178
179        let key = BatchKey::from_metadata(&metadata);
180        let mut guard = self.inner.lock();
181        let max_pending_bytes = self
182            .config
183            .max_bytes
184            .saturating_mul(8)
185            .max(self.config.max_bytes);
186
187        let prospective_total = guard.total_bytes.saturating_add(approx_bytes);
188        if prospective_total > max_pending_bytes {
189            anyhow::bail!(
190                "backpressure: buffered batches exceed limit ({} > {})",
191                prospective_total,
192                max_pending_bytes
193            );
194        }
195
196        // Scope the mutable borrow to avoid holding it across flush/remove.
197        let flush_now = {
198            let buffered = guard
199                .batches
200                .entry(key.clone())
201                .or_insert_with(|| BufferedBatch::new(&metadata));
202            buffered.add_batches(batches, &metadata, approx_bytes);
203            buffered.should_flush(&self.config)
204        };
205
206        guard.total_bytes = prospective_total;
207
208        let mut completed = Vec::new();
209        if flush_now {
210            let batch = guard
211                .batches
212                .remove(&key)
213                .ok_or_else(|| anyhow!("batch evicted before flush: {:?}", key))?;
214            guard.total_bytes = guard.total_bytes.saturating_sub(batch.total_bytes());
215            completed.push(batch.finalize()?);
216        }
217
218        drop(guard);
219
220        Ok((completed, metadata))
221    }
222
223    pub fn drain_expired(&self) -> Result<Vec<CompletedBatch<P::Metadata>>> {
224        let mut guard = self.inner.lock();
225        let mut completed = Vec::new();
226        let keys: Vec<BatchKey> = guard
227            .batches
228            .iter()
229            .filter(|(_, batch)| batch.should_flush(&self.config))
230            .map(|(key, _)| key.clone())
231            .collect();
232
233        for key in keys {
234            if let Some(batch) = guard.batches.remove(&key) {
235                guard.total_bytes = guard.total_bytes.saturating_sub(batch.total_bytes());
236                completed.push(batch.finalize()?);
237            }
238        }
239
240        Ok(completed)
241    }
242
243    pub fn drain_all(&self) -> Result<Vec<CompletedBatch<P::Metadata>>> {
244        let mut guard = self.inner.lock();
245        let drained: Vec<_> = guard.batches.drain().collect();
246        guard.total_bytes = 0;
247        drop(guard);
248
249        drained
250            .into_iter()
251            .map(|(_, batch)| batch.finalize())
252            .collect()
253    }
254}

263    fn create_test_batch(service_name: &str, record_count: usize) -> PartitionedBatch {
264        let schema = StdArc::new(Schema::new(vec![
265            Field::new(
266                "timestamp",
267                DataType::Timestamp(TimeUnit::Millisecond, None),
268                false,
269            ),
270            Field::new("service_name", DataType::Utf8, true),
271            Field::new("severity_number", DataType::Int64, true),
272        ]));
273
274        let timestamps: Vec<i64> = (0..record_count)
275            .map(|i| 1_700_000_000_000 + i as i64)
276            .collect();
277        let services: Vec<&str> = vec![service_name; record_count];
278        let severities: Vec<i64> = vec![9; record_count];
279
280        let batch = RecordBatch::try_new(
281            schema,
282            vec![
283                StdArc::new(TimestampMillisecondArray::from(timestamps.clone())),
284                StdArc::new(StringArray::from(services)),
285                StdArc::new(Int64Array::from(severities)),
286            ],
287        )
288        .unwrap();
289
290        PartitionedBatch {
291            batch,
292            service_name: Arc::from(service_name),
293            min_timestamp_micros: timestamps[0] * 1000, // Convert ms to us
294            record_count,
295        }
296    }

298    #[test]
299    fn test_batch_manager_accumulation() {
300        let config = BatchConfig {
301            max_rows: 100,
302            max_bytes: 1024 * 1024,
303            max_age: Duration::from_secs(10),
304        };
305        let manager = BatchManager::<LogSignalProcessor>::new(config);
306
307        // First request - should not flush
308        let request1 = create_test_batch("test-service", 10);
309        let approx1 = 320; // Approximate bytes
310        let (completed1, _meta1) = manager.ingest(&request1, approx1).unwrap();
311        assert_eq!(completed1.len(), 0); // Not flushed yet
312
313        // Second request - should not flush (total 20 rows)
314        let request2 = create_test_batch("test-service", 10);
315        let approx2 = 320;
316        let (completed2, _meta2) = manager.ingest(&request2, approx2).unwrap();
317        assert_eq!(completed2.len(), 0); // Still not flushed
318
319        // Third test with smaller limit - should flush when hitting threshold
320        let config_small = BatchConfig {
321            max_rows: 20,
322            max_bytes: 1024 * 1024,
323            max_age: Duration::from_secs(10),
324        };
325        let manager_small = BatchManager::<LogSignalProcessor>::new(config_small);
326
327        let req1 = create_test_batch("test-service", 10);
328        let approx_small_1 = 320;
329        let (c1, _) = manager_small.ingest(&req1, approx_small_1).unwrap();
330        assert_eq!(c1.len(), 0); // 10 rows < 20, no flush
331
332        let req2 = create_test_batch("test-service", 10);
333        let approx_small_2 = 320;
334        let (c2, _) = manager_small.ingest(&req2, approx_small_2).unwrap();
335        assert_eq!(c2.len(), 1); // 10 + 10 = 20 rows, should flush!
336        assert_eq!(
337            c2[0].batches.iter().map(|b| b.num_rows()).sum::<usize>(),
338            20
339        );
340    }
