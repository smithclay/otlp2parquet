32#[cfg(not(target_family = "wasm"))]
33impl AsyncFileWriter for OpendalAsyncWriter {
34    fn write(&mut self, bs: Bytes) -> BoxFuture<'_, ParquetResult<()>> {
35        Box::pin(async move {
36            self.0
37                .write(bs)
38                .await
39                .map_err(|e| ParquetError::External(Box::new(e)))?;
40            Ok(())
41        })
42    }
43
44    fn complete(&mut self) -> BoxFuture<'_, ParquetResult<()>> {
45        Box::pin(async move {
46            self.0
47                .close()
48                .await
49                .map_err(|e| ParquetError::External(Box::new(e)))?;
50            Ok(())
51        })
52    }
53}

86/// Write a batch as a Parquet file
87#[cfg(not(target_family = "wasm"))]
88async fn write_plain_parquet(
89    signal_type: SignalType,
90    metric_type: Option<&str>,
91    service_name: &str,
92    timestamp_micros: i64,
93    batch: &RecordBatch,
94) -> Result<String> {
95    // Get global storage operator
96    let op = crate::storage::get_operator().ok_or_else(|| {
97        WriterError::write_failure(
98            "Storage operator not initialized. Call initialize_storage() with RuntimeConfig before writing."
99                .to_string(),
100        )
101    })?;
102
103    // Generate timestamped file path with partitioning
104    // Format: {signal}/{service}/year={year}/month={month}/day={day}/hour={hour}/{uuid}.parquet
105    let file_path =
106        generate_parquet_path(signal_type, metric_type, service_name, timestamp_micros)?;
107
108    tracing::debug!("Writing plain Parquet to path: {}", file_path);
109
110    // Initialize streaming writer
111    let op_writer = op.writer(&file_path).await.map_err(|e| {
112        WriterError::write_failure(format!("Failed to open writer for '{}': {}", file_path, e))
113    })?;
114    let async_writer = OpendalAsyncWriter::new(op_writer);
115
116    let mut parquet_writer = AsyncArrowWriter::try_new(
117        async_writer,
118        batch.schema(),
119        Some(otlp2parquet_core::parquet::encoding::writer_properties().clone()),
120    )
121    .map_err(|e| WriterError::write_failure(format!("Failed to create Parquet writer: {}", e)))?;
122
123    parquet_writer
124        .write(batch)
125        .await
126        .map_err(|e| WriterError::write_failure(format!("Failed to write RecordBatch: {}", e)))?;
127
128    parquet_writer.finish().await.map_err(|e| {
129        WriterError::write_failure(format!("Failed to close Parquet writer: {}", e))
130    })?;
131
132    let bytes_written = parquet_writer.bytes_written();
133
134    let row_count = batch.num_rows();
135    tracing::info!(
136        "✓ Wrote {} rows to '{}' (plain Parquet, {} bytes)",
137        row_count,
138        file_path,
139        bytes_written
140    );
141
142    Ok(file_path)
143}

145/// 48MB - WASM environments typically have ~128MB memory limit.
146/// Parquet writing requires buffer + Arrow arrays in memory simultaneously.
147/// 48MB leaves ~80MB for Arrow buffers, runtime, and other allocations.
148#[cfg(target_family = "wasm")]
149const WASM_MAX_BUFFER_BYTES: usize = 48 * 1024 * 1024;
150
151/// WASM plain Parquet path: write into an in-memory buffer synchronously to avoid Send bounds.
152#[cfg(target_family = "wasm")]
153async fn write_plain_parquet(
154    signal_type: SignalType,
155    metric_type: Option<&str>,
156    service_name: &str,
157    timestamp_micros: i64,
158    batch: &RecordBatch,
159) -> Result<String> {
160    // Guard against OOM: estimate batch size and reject if too large for WASM buffer
161    let estimated_size = batch.get_array_memory_size();
162    if estimated_size > WASM_MAX_BUFFER_BYTES {
163        return Err(WriterError::write_failure(format!(
164            "Batch too large for WASM memory: {} bytes (limit: {} bytes, {} rows). \
165             Reduce batch size or use an upstream OTel Collector to split requests.",
166            estimated_size,
167            WASM_MAX_BUFFER_BYTES,
168            batch.num_rows()
169        )));
170    }
171
172    let op = crate::storage::get_operator().ok_or_else(|| {
173        WriterError::write_failure(
174            "Storage operator not initialized. Call initialize_storage() with RuntimeConfig before writing."
175                .to_string(),
176        )
177    })?;
178
179    let file_path =
180        generate_parquet_path(signal_type, metric_type, service_name, timestamp_micros)?;
181
182    tracing::debug!("Writing plain Parquet (WASM) to path: {}", file_path);
183
184    let mut buffer = Vec::new();
185    {
186        let mut writer = ArrowWriter::try_new(
187            &mut buffer,
188            batch.schema(),
189            Some(otlp2parquet_core::parquet::encoding::writer_properties().clone()),
190        )
191        .map_err(|e| {
192            WriterError::write_failure(format!("Failed to create Parquet writer: {}", e))
193        })?;
194        writer.write(batch).map_err(|e| {
195            WriterError::write_failure(format!("Failed to write RecordBatch: {}", e))
196        })?;
197        writer.close().map_err(|e| {
198            WriterError::write_failure(format!("Failed to close Parquet writer: {}", e))
199        })?;
200    }
201
202    let bytes_written = buffer.len();
203    op.write(&file_path, buffer).await.map_err(|e| {
204        WriterError::write_failure(format!(
205            "Failed to upload Parquet to '{}': {}",
206            file_path, e
207        ))
208    })?;
209
210    let row_count = batch.num_rows();
211    tracing::info!(
212        "✓ Wrote {} rows to '{}' (plain Parquet, {} bytes, wasm buffer)",
213        row_count,
214        file_path,
215        bytes_written
216    );
217
218    Ok(file_path)
219}

221/// WASM plain Parquet path: write multiple batches as separate row groups.
222///
223/// This achieves zero-copy by writing each batch as a separate Parquet row group,
224/// avoiding the full data copy that concat_batches performs.
225#[cfg(target_family = "wasm")]
226async fn write_plain_parquet_multi(
227    signal_type: SignalType,
228    metric_type: Option<&str>,
229    service_name: &str,
230    timestamp_micros: i64,
231    batches: &[RecordBatch],
232) -> Result<String> {
233    if batches.is_empty() {
234        return Err(WriterError::write_failure(
235            "No batches to write".to_string(),
236        ));
237    }
238
239    // Guard against OOM: estimate total size and reject if too large for WASM buffer
240    let estimated_size: usize = batches.iter().map(|b| b.get_array_memory_size()).sum();
241    if estimated_size > WASM_MAX_BUFFER_BYTES {
242        return Err(WriterError::write_failure(format!(
243            "Batches too large for WASM memory: {} bytes (limit: {} bytes, {} batches). \
244             Reduce batch size or use an upstream OTel Collector to split requests.",
245            estimated_size,
246            WASM_MAX_BUFFER_BYTES,
247            batches.len()
248        )));
249    }
250
251    let op = crate::storage::get_operator().ok_or_else(|| {
252        WriterError::write_failure(
253            "Storage operator not initialized. Call initialize_storage() with RuntimeConfig before writing."
254                .to_string(),
255        )
256    })?;
257
258    let file_path =
259        generate_parquet_path(signal_type, metric_type, service_name, timestamp_micros)?;
260
261    tracing::debug!(
262        "Writing {} batches as separate row groups (WASM) to path: {}",
263        batches.len(),
264        file_path
265    );
266
267    // Validate all batches have the same schema
268    let expected_schema = batches[0].schema();
269    for (i, batch) in batches.iter().enumerate().skip(1) {
270        if batch.schema() != expected_schema {
271            return Err(WriterError::write_failure(format!(
272                "Schema mismatch in batch {}: expected {} fields, got {}. \
273                 All batches must have identical schemas.",
274                i,
275                expected_schema.fields().len(),
276                batch.schema().fields().len()
277            )));
278        }
279    }
280
281    let mut buffer = Vec::new();
282    let total_rows: usize;
283    {
284        let mut writer = ArrowWriter::try_new(
285            &mut buffer,
286            expected_schema,
287            Some(otlp2parquet_core::parquet::encoding::writer_properties().clone()),
288        )
289        .map_err(|e| {
290            WriterError::write_failure(format!("Failed to create Parquet writer: {}", e))
291        })?;
292
293        // Write each batch as a separate row group
294        for batch in batches {
295            writer.write(batch).map_err(|e| {
296                WriterError::write_failure(format!("Failed to write RecordBatch: {}", e))
297            })?;
298        }
299
300        total_rows = batches.iter().map(|b| b.num_rows()).sum();
301
302        writer.close().map_err(|e| {
303            WriterError::write_failure(format!("Failed to close Parquet writer: {}", e))
304        })?;
305    }
306
307    let bytes_written = buffer.len();
308    op.write(&file_path, buffer).await.map_err(|e| {
309        WriterError::write_failure(format!(
310            "Failed to upload Parquet to '{}': {}",
311            file_path, e
312        ))
313    })?;
314
315    tracing::info!(
316        "✓ Wrote {} rows ({} batches, {} row groups) to '{}' (plain Parquet, {} bytes, wasm buffer)",
317        total_rows,
318        batches.len(),
319        batches.len(),
320        file_path,
321        bytes_written
322    );
323
324    Ok(file_path)
325}

327pub async fn write_batch(req: WriteBatchRequest<'_>) -> Result<String> {
328    let row_count = req.batch.num_rows();
329
330    tracing::debug!(
331        "Writing {} rows (service: {}, signal: {:?}, metric: {:?})",
332        row_count,
333        req.service_name,
334        req.signal_type,
335        req.metric_type
336    );
337
338    write_plain_parquet(
339        req.signal_type,
340        req.metric_type,
341        req.service_name,
342        req.timestamp_micros,
343        req.batch,
344    )
345    .await
346}

374/// Generate a partitioned file path for plain Parquet files
375///
376/// Format: {prefix?}{signal_type}/{service}/year={year}/month={month}/day={day}/hour={hour}/{timestamp}-{uuid}.parquet
377/// Example: logs/my-service/year=2025/month=01/day=15/hour=10/1736938800000000-<uuid>.parquet
378/// Example with prefix: smoke-abc123/logs/my-service/year=2025/month=01/day=15/hour=10/1736938800000000-<uuid>.parquet
379fn generate_parquet_path(
380    signal_type: SignalType,
381    metric_type: Option<&str>,
382    service_name: &str,
383    timestamp_micros: i64,
384) -> Result<String> {
385    let (year, month, day, hour) = partition_from_timestamp(timestamp_micros);
386
387    let signal_prefix: Cow<'_, str> = match signal_type {
388        SignalType::Logs => Cow::Borrowed("logs"),
389        SignalType::Traces => Cow::Borrowed("traces"),
390        SignalType::Metrics => {
391            if let Some(mtype) = metric_type {
392                Cow::Owned(format!("metrics/{}", mtype))
393            } else {
394                Cow::Borrowed("metrics")
395            }
396        }
397    };
398
399    let safe_service = sanitize_service_name(service_name);
400    let suffix = Uuid::new_v4().simple();
401
402    // Get storage prefix if configured (e.g., "smoke-abc123/")
403    let storage_prefix = crate::storage::get_storage_prefix().unwrap_or("");
404
405    Ok(format!(
406        "{}{}/{}/year={}/month={:02}/day={:02}/hour={:02}/{}-{}.parquet",
407        storage_prefix,
408        signal_prefix,
409        safe_service,
410        year,
411        month,
412        day,
413        hour,
414        timestamp_micros,
415        suffix
416    ))
417}
418
419fn sanitize_service_name(service_name: &str) -> Cow<'_, str> {
420    // Characters that are invalid or problematic in object storage paths
421    const INVALID: [char; 10] = ['/', '\\', ' ', ':', '*', '?', '"', '<', '>', '|'];
422
423    if service_name.is_empty() {
424        return Cow::Borrowed("unknown-service");
425    }
426
427    if service_name.chars().any(|c| INVALID.contains(&c)) {
428        let sanitized = service_name
429            .chars()
430            .map(|c| if INVALID.contains(&c) { '_' } else { c })
431            .collect::<String>();
432        Cow::Owned(sanitized)
433    } else {
434        Cow::Borrowed(service_name)
435    }
436}

438/// Get fallback timestamp partition values when timestamp_micros is invalid.
439/// Uses platform-appropriate time source (std::time on native, js_sys on WASM).
440#[cfg(not(target_family = "wasm"))]
441fn fallback_partition() -> (i32, u8, u8, u8) {
442    let now = OffsetDateTime::now_utc();
443    (now.year(), u8::from(now.month()), now.day(), now.hour())
444}
445
446#[cfg(target_family = "wasm")]
447fn fallback_partition() -> (i32, u8, u8, u8) {
448    // js_sys::Date::now() returns milliseconds since epoch
449    let now_ms = js_sys::Date::now() as i64;
450    let nanos = i128::from(now_ms).saturating_mul(1_000_000);
451    match OffsetDateTime::from_unix_timestamp_nanos(nanos) {
452        Ok(dt) => (dt.year(), u8::from(dt.month()), dt.day(), dt.hour()),
453        Err(_) => {
454            // Absolute fallback if even JS time conversion fails
455            (2025, 1, 1, 0)
456        }
457    }
458}
459
460fn partition_from_timestamp(timestamp_micros: i64) -> (i32, u8, u8, u8) {
461    if timestamp_micros <= 0 {
462        return fallback_partition();
463    }
464
465    let nanos = i128::from(timestamp_micros).saturating_mul(1_000);
466    match OffsetDateTime::from_unix_timestamp_nanos(nanos) {
467        Ok(dt) => (dt.year(), u8::from(dt.month()), dt.day(), dt.hour()),
468        Err(_) => fallback_partition(),
469    }
470}
