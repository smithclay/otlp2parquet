1//! Storage operator initialization and management
2//!
3//! Provides a global operator for direct Parquet writes when catalog is not used.
4
5use once_cell::sync::OnceCell;
6use otlp2parquet_core::config::{RuntimeConfig, StorageBackend};
7
8static OPERATOR: OnceCell<opendal::Operator> = OnceCell::new();
9
10/// Initialize storage operator from RuntimeConfig
11///
12/// This creates an OpenDAL operator that can be used for direct Parquet writes
13/// when catalog is not configured. The operator is stored globally and reused.
14pub fn initialize_storage(config: &RuntimeConfig) -> crate::Result<()> {
15    if OPERATOR.get().is_some() {
16        return Ok(()); // Already initialized
17    }
18
19    let operator = match config.storage.backend {
20        #[cfg(not(target_arch = "wasm32"))]
21        StorageBackend::Fs => {
22            let fs = config.storage.fs.as_ref().ok_or_else(|| {
23                crate::WriterError::write_failure(
24                    "fs config required for filesystem backend".to_string(),
25                )
26            })?;
27
28            let fs_builder = opendal::services::Fs::default().root(&fs.path);
29            opendal::Operator::new(fs_builder)
30                .map_err(|e| {
31                    crate::WriterError::write_failure(format!(
32                        "Failed to create filesystem operator: {}",
33                        e
34                    ))
35                })?
36                .finish()
37        }
38        #[cfg(target_arch = "wasm32")]
39        StorageBackend::Fs => {
40            return Err(crate::WriterError::write_failure(
41                "Filesystem storage backend not supported in WASM environments".to_string(),
42            ));
43        }
44        StorageBackend::S3 => {
45            let s3 = config.storage.s3.as_ref().ok_or_else(|| {
46                crate::WriterError::write_failure("s3 config required for S3 backend".to_string())
47            })?;
48
49            let mut s3_builder = opendal::services::S3::default()
50                .bucket(&s3.bucket)
51                .region(&s3.region);
52
53            if let Some(endpoint) = &s3.endpoint {
54                s3_builder = s3_builder.endpoint(endpoint);
55            }
56
57            opendal::Operator::new(s3_builder)
58                .map_err(|e| {
59                    crate::WriterError::write_failure(format!(
60                        "Failed to create S3 operator: {}",
61                        e
62                    ))
63                })?
64                .finish()
65        }
66        StorageBackend::R2 => {
67            let r2 = config.storage.r2.as_ref().ok_or_else(|| {
68                crate::WriterError::write_failure("r2 config required for R2 backend".to_string())
69            })?;
70
71            // Use endpoint from config if provided, otherwise construct from account_id
72            let endpoint = r2
73                .endpoint
74                .clone()
75                .unwrap_or_else(|| format!("https://{}.r2.cloudflarestorage.com", r2.account_id));
76
77            let r2_builder = opendal::services::S3::default()
78                .bucket(&r2.bucket)
79                .region("auto")
80                .endpoint(&endpoint)
81                .access_key_id(&r2.access_key_id)
82                .secret_access_key(&r2.secret_access_key);
83
84            opendal::Operator::new(r2_builder)
85                .map_err(|e| {
86                    crate::WriterError::write_failure(format!(
87                        "Failed to create R2 operator: {}",
88                        e
89                    ))
90                })?
91                .finish()
92        }
93    };
94
95    match OPERATOR.set(operator) {
96        Ok(_) => Ok(()),
97        Err(_) => Ok(()), // another thread initialized first
98    }
99}
100
101/// Get the global storage operator
102///
103/// Returns None if operator has not been initialized via `initialize_storage`.
104pub(crate) fn get_operator() -> Option<&'static opendal::Operator> {
105    OPERATOR.get()
106}
