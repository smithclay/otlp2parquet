39/// Application state shared across all requests
40#[derive(Clone)]
41pub(crate) struct AppState {
42    pub storage: Arc<OpenDalStorage>,
43    pub parquet_writer: Arc<ParquetWriter>,
44    pub batcher: Option<Arc<BatchManager>>,
45    pub passthrough: PassthroughBatcher,
46    pub max_payload_bytes: usize,
47}
48
49/// Error type that implements IntoResponse
50pub(crate) struct AppError {
51    status: StatusCode,
52    error: anyhow::Error,
53}
54
55impl IntoResponse for AppError {
56    fn into_response(self) -> Response {
57        error!("Request error: {:?}", self.error);
58        (
59            self.status,
60            Json(json!({
61                "error": self.error.to_string(),
62            })),
63        )
64            .into_response()
65    }
66}
67
68impl<E> From<E> for AppError
69where
70    E: Into<anyhow::Error>,
71{
72    fn from(err: E) -> Self {
73        Self {
74            status: StatusCode::INTERNAL_SERVER_ERROR,
75            error: err.into(),
76        }
77    }
78}
79
80impl AppError {
81    pub fn with_status(status: StatusCode, error: anyhow::Error) -> Self {
82        Self { status, error }
83    }
84}

86/// Graceful shutdown handler
87async fn shutdown_signal() {
88    let ctrl_c = async {
89        signal::ctrl_c()
90            .await
91            .expect("failed to install Ctrl+C handler");
92    };
93
94    #[cfg(unix)]
95    let terminate = async {
96        signal::unix::signal(signal::unix::SignalKind::terminate())
97            .expect("failed to install signal handler")
98            .recv()
99            .await;
100    };
101
102    #[cfg(not(unix))]
103    let terminate = std::future::pending::<()>();
104
105    tokio::select! {
106        _ = ctrl_c => {
107            info!("Received Ctrl+C, starting graceful shutdown...");
108        },
109        _ = terminate => {
110            info!("Received SIGTERM, starting graceful shutdown...");
111        },
112    }
113}

115/// Entry point for server mode
116pub async fn run() -> Result<()> {
117    // Load configuration
118    let config = RuntimeConfig::load().context("Failed to load configuration")?;
119
120    // Initialize tracing with config
121    init_tracing(&config);
122
123    // Configure Parquet writer properties before first use
124    set_parquet_row_group_size(config.storage.parquet_row_group_size);
125
126    info!("Server mode - full-featured HTTP server with multi-backend storage");
127
128    // Get listen address from config
129    let addr = config
130        .server
131        .as_ref()
132        .expect("server config required")
133        .listen_addr
134        .clone();
135
136    // Initialize storage
137    let storage = init_storage(&config)?;
138    let parquet_writer = Arc::new(ParquetWriter::new(storage.operator().clone()));
139
140    // Configure batching
141    let batch_config = BatchConfig {
142        max_rows: config.batch.max_rows,
143        max_bytes: config.batch.max_bytes,
144        max_age: Duration::from_secs(config.batch.max_age_secs),
145    };
146
147    let batcher = if !config.batch.enabled {
148        info!("Batching disabled by configuration");
149        None
150    } else {
151        info!(
152            "Batching enabled (max_rows={} max_bytes={} max_age={}s)",
153            batch_config.max_rows,
154            batch_config.max_bytes,
155            batch_config.max_age.as_secs()
156        );
157        Some(Arc::new(BatchManager::new(batch_config)))
158    };
159
160    let max_payload_bytes = config.request.max_payload_bytes;
161    info!("Max payload size set to {} bytes", max_payload_bytes);
162
163    // Create app state
164    let state = AppState {
165        storage,
166        parquet_writer,
167        batcher,
168        passthrough: PassthroughBatcher::default(),
169        max_payload_bytes,
170    };
171
172    // Build router
173    let app = Router::new()
174        .route("/v1/logs", post(handle_logs))
175        .route("/v1/traces", post(handle_traces))
176        .route("/v1/metrics", post(handle_metrics))
177        .route("/health", get(health_check))
178        .route("/ready", get(ready_check))
179        .with_state(state);
180
181    // Create TCP listener
182    let listener = tokio::net::TcpListener::bind(&addr)
183        .await
184        .context(format!("Failed to bind to {}", addr))?;
185
186    info!("OTLP HTTP endpoint listening on http://{}", addr);
187    info!("Routes:");
188    info!("  POST http://{}/v1/logs    - OTLP log ingestion", addr);
189    info!("  POST http://{}/v1/metrics - OTLP metrics ingestion", addr);
190    info!("  POST http://{}/v1/traces  - OTLP trace ingestion", addr);
191    info!("  GET  http://{}/health     - Health check", addr);
192    info!("  GET  http://{}/ready      - Readiness check", addr);
193    info!("Press Ctrl+C or send SIGTERM to stop");
194
195    // Start server with graceful shutdown
196    axum::serve(listener, app)
197        .with_graceful_shutdown(shutdown_signal())
198        .await
199        .context("Server error")?;
200
201    info!("Server shutdown complete");
202
203    Ok(())
204}
