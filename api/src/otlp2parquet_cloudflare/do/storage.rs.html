12/// Initialize SQLite schema (idempotent).
13///
14/// Creates tables for batch storage and DO state if they don't exist.
15/// Runs on every DO wake to ensure schema is present.
16pub fn init_schema(state: &State) -> Result<()> {
17    let sql = state.storage().sql();
18
19    // Check if batches table has new schema (chunk_group_id column)
20    // If old schema exists, drop it (one-time migration, data loss acceptable per design)
21    let has_new_schema = sql
22        .exec("SELECT chunk_group_id FROM batches LIMIT 1", None)
23        .is_ok();
24    if !has_new_schema {
25        if let Err(e) = sql.exec("DROP TABLE IF EXISTS batches", None) {
26            tracing::warn!(error = %e, "Failed to drop old batches table during migration");
27        }
28    }
29
30    // Create batches table with chunking support
31    sql.exec(
32        "CREATE TABLE IF NOT EXISTS batches (
33            id INTEGER PRIMARY KEY AUTOINCREMENT,
34            chunk_group_id TEXT NOT NULL,
35            chunk_index INTEGER NOT NULL,
36            blob BLOB NOT NULL,
37            bytes INTEGER NOT NULL,
38            rows INTEGER NOT NULL
39        )",
40        None,
41    )
42    .map_err(|e| worker::Error::RustError(format!("Failed to create batches table: {}", e)))?;
43
44    // Index for efficient chunk group queries (reassembly and FIFO ordering)
45    sql.exec(
46        "CREATE INDEX IF NOT EXISTS idx_batches_chunk_group ON batches(chunk_group_id)",
47        None,
48    )
49    .map_err(|e| worker::Error::RustError(format!("Failed to create batches index: {}", e)))?;
50
51    sql.exec(
52        "CREATE TABLE IF NOT EXISTS state (
53            id INTEGER PRIMARY KEY CHECK (id = 1),
54            signal_type TEXT,
55            service_name TEXT,
56            first_event_timestamp INTEGER
57        )",
58        None,
59    )
60    .map_err(|e| worker::Error::RustError(format!("Failed to create state table: {}", e)))?;
61
62    sql.exec("INSERT OR IGNORE INTO state (id) VALUES (1)", None)
63        .map_err(|e| worker::Error::RustError(format!("Failed to initialize state row: {}", e)))?;
64
65    Ok(())
66}

68/// Store a batch blob in SQLite, splitting into chunks if necessary.
69///
70/// Large blobs are split into MAX_CHUNK_BYTES (512KB) chunks to avoid
71/// SQLite's blob size limit (~2MB in Cloudflare DO). All chunks share
72/// the same `chunk_group_id` and are ordered by `chunk_index`.
73pub fn store_batch(state: &State, blob: &[u8], rows: i64) -> Result<()> {
74    let chunk_group_id = uuid::Uuid::new_v4().to_string();
75    let sql = state.storage().sql();
76
77    for (chunk_index, chunk) in blob.chunks(MAX_CHUNK_BYTES).enumerate() {
78        sql.exec(
79            "INSERT INTO batches (chunk_group_id, chunk_index, blob, bytes, rows) VALUES (?, ?, ?, ?, ?)",
80            Some(vec![
81                SqlStorageValue::String(chunk_group_id.clone()),
82                SqlStorageValue::Integer(chunk_index as i64),
83                SqlStorageValue::Blob(chunk.to_vec()),
84                SqlStorageValue::Integer(chunk.len() as i64),
85                SqlStorageValue::Integer(rows),
86            ]),
87        )?;
88    }
89    Ok(())
90}

92/// Get total bytes and rows of pending batches.
93///
94/// Bytes are summed across all chunks. Rows are counted once per chunk group
95/// (since the same row count is stored on each chunk within a group).
96pub fn get_batch_totals(state: &State) -> Result<(i64, i64)> {
97    let sql = state.storage().sql();
98    // Sum all chunk bytes, but only count rows once per group (chunk_index = 0)
99    let cursor = sql.exec(
100        "SELECT COALESCE(SUM(bytes), 0) as total_bytes,
101                COALESCE(SUM(CASE WHEN chunk_index = 0 THEN rows ELSE 0 END), 0) as total_rows
102         FROM batches",
103        None,
104    )?;
105    #[derive(Deserialize)]
106    struct Totals {
107        total_bytes: i64,
108        total_rows: i64,
109    }
110    let row: Totals = cursor
111        .one()
112        .map_err(|e| worker::Error::RustError(format!("Failed to get batch totals: {}", e)))?;
113    Ok((row.total_bytes, row.total_rows))
114}

116/// Get count of pending batch groups (not individual chunks).
117pub fn get_batch_count(state: &State) -> Result<i64> {
118    let sql = state.storage().sql();
119    let cursor = sql.exec(
120        "SELECT COUNT(DISTINCT chunk_group_id) as cnt FROM batches",
121        None,
122    )?;
123    #[derive(Deserialize)]
124    struct Count {
125        cnt: i64,
126    }
127    let row: Count = cursor
128        .one()
129        .map_err(|e| worker::Error::RustError(format!("Failed to get batch count: {}", e)))?;
130    Ok(row.cnt)

133/// Load batch groups for flush, up to the specified byte limit (FIFO order).
134///
135/// Returns the oldest batch groups that fit within `max_bytes`. Each group's
136/// chunks are reassembled into the original IPC blob. Groups are NOT deleted
137/// from SQLite - call `delete_batch_groups()` after successful write.
138pub fn load_batches_for_flush(state: &State, max_bytes: usize) -> Result<Vec<BatchGroup>> {
139    let sql = state.storage().sql();
140
141    // Step 1: Get group metadata ordered by FIFO (earliest id = oldest)
142    #[derive(Deserialize)]
143    struct GroupMeta {
144        chunk_group_id: String,
145        #[allow(dead_code)]
146        first_id: i64,
147        total_bytes: i64,
148        rows: i64,
149    }
150
151    let groups_cursor = sql.exec(
152        "SELECT chunk_group_id, MIN(id) as first_id, SUM(bytes) as total_bytes, MAX(rows) as rows
153         FROM batches
154         GROUP BY chunk_group_id
155         ORDER BY first_id",
156        None,
157    )?;
158
159    let groups: Vec<GroupMeta> = groups_cursor.to_array()?.into_iter().collect();
160
161    // Step 2: Select groups until we hit max_bytes (FIFO)
162    let mut selected_groups: Vec<GroupMeta> = Vec::new();
163    let mut accumulated_bytes: usize = 0;
164
165    for group in groups {
166        let group_bytes = group.total_bytes as usize;
167        // Always include at least one group (even if it exceeds max_bytes)
168        // to avoid stalling if a single batch is larger than the limit
169        if accumulated_bytes + group_bytes > max_bytes && !selected_groups.is_empty() {
170            break;
171        }
172        accumulated_bytes += group_bytes;
173        selected_groups.push(group);
174    }
175
176    if selected_groups.is_empty() {
177        return Ok(vec![]);
178    }
179
180    // Step 3: Load and reassemble chunks for selected groups
181    let mut batch_groups = Vec::with_capacity(selected_groups.len());
182
183    for group_meta in selected_groups {
184        let chunks_cursor = sql.exec(
185            "SELECT blob, bytes FROM batches WHERE chunk_group_id = ? ORDER BY chunk_index",
186            Some(vec![SqlStorageValue::String(
187                group_meta.chunk_group_id.clone(),
188            )]),
189        )?;
190
191        // Reassemble chunks using raw() iterator for proper blob handling
192        let mut reassembled = Vec::new();
193        let mut total_bytes: usize = 0;
194
195        for row_result in chunks_cursor.raw() {
196            let row = row_result.map_err(|e| {
197                worker::Error::RustError(format!("Failed to read chunk row: {}", e))
198            })?;
199
200            let mut iter = row.into_iter();
201            if let Some(SqlStorageValue::Blob(chunk_data)) = iter.next() {
202                total_bytes += chunk_data.len();
203                reassembled.extend_from_slice(&chunk_data);
204            }
205        }
206
207        batch_groups.push(BatchGroup {
208            chunk_group_id: group_meta.chunk_group_id,
209            reassembled_blob: Bytes::from(reassembled),
210            total_bytes,
211            rows: group_meta.rows,
212        });
213    }
214
215    Ok(batch_groups)
216}

218/// Delete specific batch groups from SQLite after successful flush.
219pub fn delete_batch_groups(state: &State, groups: &[BatchGroup]) -> Result<()> {
220    let sql = state.storage().sql();
221    for group in groups {
222        sql.exec(
223            "DELETE FROM batches WHERE chunk_group_id = ?",
224            Some(vec![SqlStorageValue::String(group.chunk_group_id.clone())]),
225        )?;
226    }
227    Ok(())
228}

230/// Get persistent DO state from SQLite.
231pub fn get_do_state(state: &State) -> Result<DoState> {
232    let sql = state.storage().sql();
233    let cursor = sql.exec(
234        "SELECT signal_type, service_name, first_event_timestamp FROM state WHERE id = 1",
235        None,
236    )?;
237    #[derive(Deserialize, Default)]
238    struct StateRow {
239        signal_type: Option<String>,
240        service_name: Option<String>,
241        first_event_timestamp: Option<i64>,
242    }
243    let row: StateRow = cursor.one().unwrap_or_else(|e| {
244        tracing::debug!(error = %e, "No DO state found, using defaults");
245        StateRow::default()
246    });
247    Ok(DoState {
248        signal_type: row.signal_type,
249        service_name: row.service_name,
250        first_event_timestamp: row.first_event_timestamp,
251    })
252}

254/// Set signal type and service name (identity) in SQLite.
255pub fn set_identity(state: &State, signal_type: &str, service_name: &str) -> Result<()> {
256    let sql = state.storage().sql();
257    sql.exec(
258        "UPDATE state SET signal_type = ?, service_name = ? WHERE id = 1",
259        Some(vec![
260            SqlStorageValue::String(signal_type.to_string()),
261            SqlStorageValue::String(service_name.to_string()),
262        ]),
263    )?;
264    Ok(())
265}

267/// Update first event timestamp (only if earlier than existing).
268pub fn update_first_event_timestamp(state: &State, timestamp_micros: i64) -> Result<()> {
269    let sql = state.storage().sql();
270    sql.exec(
271        "UPDATE state SET first_event_timestamp = ?
272         WHERE id = 1 AND (first_event_timestamp IS NULL OR first_event_timestamp > ?)",
273        Some(vec![
274            SqlStorageValue::Integer(timestamp_micros),
275            SqlStorageValue::Integer(timestamp_micros),
276        ]),
277    )?;
278    Ok(())
279}
