11/// Handle logs request
12pub async fn handle_logs_request(
13    body_bytes: &[u8],
14    format: otlp2parquet_core::InputFormat,
15    content_type: Option<&str>,
16    storage: &Arc<otlp2parquet_storage::opendal_storage::OpenDalStorage>,
17) -> Result<Response> {
18    // Cloudflare Workers: Always use passthrough (no batching)
19    // Batching requires time-based operations which aren't supported in WASM
20    let passthrough: PassthroughBatcher<LogSignalProcessor> = PassthroughBatcher::default();
21
22    // Process logs
23    let request = otlp::parse_otlp_request(body_bytes, format).map_err(|e| {
24        console_error!(
25            "Failed to parse OTLP logs (format: {:?}, content-type: {:?}): {:?}",
26            format,
27            content_type,
28            e
29        );
30        worker::Error::RustError(format!("Processing error: {}", e))
31    })?;
32
33    // Convert directly to Arrow (no batching)
34    let batch = passthrough.ingest(&request).map_err(|err| {
35        console_error!("Failed to convert OTLP to Arrow: {:?}", err);
36        worker::Error::RustError(format!("Encoding error: {}", err))
37    })?;
38
39    let metadata = batch.metadata.clone();
40    let uploads = vec![batch];
41
42    let mut uploaded_paths = Vec::new();
43    for batch in uploads {
44        // WASM-specific: Write Parquet synchronously, then upload via OpenDAL
45        // (parquet_opendal doesn't work in WASM due to tokio::spawn requiring Send)
46        let parquet_bytes = parquet::write_batches_to_parquet(&batch.batches).map_err(|e| {
47            console_error!("Failed to serialize Parquet: {:?}", e);
48            worker::Error::RustError(format!("Parquet serialization error: {}", e))
49        })?;
50
51        // Compute Blake3 hash for content-addressable storage
52        let hash_bytes = blake3::hash(&parquet_bytes);
53        let hash_hex = hex::encode(hash_bytes.as_bytes());
54
55        // Generate partition path with hash
56        let partition_path = otlp2parquet_storage::partition::generate_partition_path(
57            &batch.metadata.service_name,
58            batch.metadata.first_timestamp_nanos,
59            &hash_hex,
60        );
61
62        // Upload to R2 via OpenDAL (this DOES work in WASM)
63        storage
64            .write(&partition_path, parquet_bytes)
65            .await
66            .map_err(|e| {
67                console_error!("Failed to write to R2: {:?}", e);
68                worker::Error::RustError(format!("Storage error: {}", e))
69            })?;
70
71        uploaded_paths.push(partition_path);
72    }
73
74    let response_body = json!({
75        "status": "ok",
76        "records_processed": metadata.record_count,
77        "flush_count": uploaded_paths.len(),
78        "partitions": uploaded_paths,
79    });
80
81    Response::from_json(&response_body)
82}

84/// Handle traces request
85pub async fn handle_traces_request(
86    body_bytes: &[u8],
87    format: otlp2parquet_core::InputFormat,
88    content_type: Option<&str>,
89    storage: &Arc<otlp2parquet_storage::opendal_storage::OpenDalStorage>,
90) -> Result<Response> {
91    // Parse OTLP traces request
92    let request = otlp::traces::parse_otlp_trace_request(body_bytes, format).map_err(|e| {
93        console_error!(
94            "Failed to parse OTLP traces (format: {:?}, content-type: {:?}): {:?}",
95            format,
96            content_type,
97            e
98        );
99        worker::Error::RustError(format!("Processing error: {}", e))
100    })?;
101
102    // Convert to Arrow
103    let (batches, metadata) =
104        otlp::traces::TraceArrowConverter::convert(&request).map_err(|e| {
105            console_error!("Failed to convert OTLP traces to Arrow: {:?}", e);
106            worker::Error::RustError(format!("Encoding error: {}", e))
107        })?;
108
109    if batches.is_empty() || metadata.span_count == 0 {
110        let response = Response::from_json(&json!({
111            "status": "ok",
112            "message": "No trace spans to process",
113        }))?;
114        return Ok(response);
115    }
116
117    // Write Parquet synchronously (WASM-compatible)
118    let parquet_bytes = parquet::write_batches_to_parquet(&batches).map_err(|e| {
119        console_error!("Failed to serialize traces Parquet: {:?}", e);
120        worker::Error::RustError(format!("Parquet serialization error: {}", e))
121    })?;
122
123    // Compute Blake3 hash
124    let hash_bytes = blake3::hash(&parquet_bytes);
125    let hash_hex = hex::encode(hash_bytes.as_bytes());
126
127    // Generate partition path for traces
128    let partition_path = otlp2parquet_storage::partition::generate_partition_path_with_signal(
129        "traces",
130        metadata.service_name.as_ref(),
131        metadata.first_timestamp_nanos,
132        &hash_hex,
133        None, // No subdirectory for traces (unlike metrics)
134    );
135
136    // Upload to R2
137    storage
138        .write(&partition_path, parquet_bytes)
139        .await
140        .map_err(|e| {
141            console_error!("Failed to write traces to R2: {:?}", e);
142            worker::Error::RustError(format!("Storage error: {}", e))
143        })?;
144
145    let response_body = json!({
146        "status": "ok",
147        "spans_processed": metadata.span_count,
148        "partitions": vec![partition_path],
149    });
150
151    Response::from_json(&response_body)
152}

154/// Handle metrics request (separate from logs due to multiple batches per type)
155pub async fn handle_metrics_request(
156    body_bytes: &[u8],
157    format: otlp2parquet_core::InputFormat,
158    content_type: Option<&str>,
159    storage: &Arc<otlp2parquet_storage::opendal_storage::OpenDalStorage>,
160) -> Result<Response> {
161    // Parse OTLP metrics request
162    let request = otlp::metrics::parse_otlp_request(body_bytes, format).map_err(|e| {
163        console_error!(
164            "Failed to parse OTLP metrics (format: {:?}, content-type: {:?}): {:?}",
165            format,
166            content_type,
167            e
168        );
169        worker::Error::RustError(format!("Processing error: {}", e))
170    })?;
171
172    // Convert to Arrow (returns multiple batches, one per metric type)
173    let converter = otlp::metrics::ArrowConverter::new();
174    let (batches_by_type, metadata) = converter.convert(request).map_err(|e| {
175        console_error!("Failed to convert OTLP metrics to Arrow: {:?}", e);
176        worker::Error::RustError(format!("Encoding error: {}", e))
177    })?;
178
179    if batches_by_type.is_empty() {
180        let response = Response::from_json(&json!({
181            "status": "ok",
182            "message": "No metrics data points to process",
183        }))?;
184        return Ok(response);
185    }
186
187    // Count total data points across all metric types
188    let total_data_points = metadata.gauge_count
189        + metadata.sum_count
190        + metadata.histogram_count
191        + metadata.exponential_histogram_count
192        + metadata.summary_count;
193
194    // Write each metric type batch to its own Parquet file
195    let mut uploaded_paths = Vec::new();
196    for (metric_type, batch) in batches_by_type {
197        let service_name = "default"; // TODO: Extract from metadata
198        let timestamp_nanos = 0; // TODO: Extract first timestamp from batch
199
200        // Write Parquet synchronously (WASM-compatible)
201        let parquet_bytes = parquet::write_batches_to_parquet(&[batch]).map_err(|e| {
202            console_error!(
203                "Failed to serialize {} metrics Parquet: {:?}",
204                metric_type,
205                e
206            );
207            worker::Error::RustError(format!("Parquet serialization error: {}", e))
208        })?;
209
210        // Compute Blake3 hash
211        let hash_bytes = blake3::hash(&parquet_bytes);
212        let hash_hex = hex::encode(hash_bytes.as_bytes());
213
214        // Generate partition path with metric type subdirectory
215        let partition_path = otlp2parquet_storage::partition::generate_partition_path_with_signal(
216            "metrics",
217            service_name,
218            timestamp_nanos,
219            &hash_hex,
220            Some(&metric_type),
221        );
222
223        // Upload to R2
224        storage
225            .write(&partition_path, parquet_bytes)
226            .await
227            .map_err(|e| {
228                console_error!("Failed to write {} metrics to R2: {:?}", metric_type, e);
229                worker::Error::RustError(format!("Storage error: {}", e))
230            })?;
231
232        uploaded_paths.push(partition_path);
233    }
234
235    let response_body = json!({
236        "status": "ok",
237        "data_points_processed": total_data_points,
238        "gauge_count": metadata.gauge_count,
239        "sum_count": metadata.sum_count,
240        "histogram_count": metadata.histogram_count,
241        "exponential_histogram_count": metadata.exponential_histogram_count,
242        "summary_count": metadata.summary_count,
243        "partitions": uploaded_paths,
244    });
245
246    Response::from_json(&response_body)
247}
