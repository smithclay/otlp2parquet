11/// Convert OtlpError to worker::Error
12fn convert_to_worker_error(err: OtlpError, request_id: &str) -> worker::Error {
13    let status_code = err.status_code();
14    let error_response = errors::ErrorResponse {
15        error: err.error_type().to_string(),
16        message: err.message(),
17        details: err.hint(),
18        request_id: Some(request_id.to_string()),
19    };
20
21    let error_json = serde_json::to_string(&error_response).unwrap_or_else(|_| {
22        r#"{"error":"internal error","code":"SERIALIZATION_FAILED"}"#.to_string()
23    });
24    worker::Error::RustError(format!("{}:{}", status_code, error_json))
25}

27/// Handle logs request
28pub async fn handle_logs_request(
29    body_bytes: &[u8],
30    format: otlp2parquet_core::InputFormat,
31    content_type: Option<&str>,
32    request_id: &str,
33) -> Result<Response> {
34    let result = process_logs_handler(body_bytes, format)
35        .await
36        .map_err(|e| {
37            tracing::error!(
38                request_id = %request_id,
39                format = ?format,
40                content_type = ?content_type,
41                error = %e.message(),
42                "Failed to process logs"
43            );
44            convert_to_worker_error(e, request_id)
45        })?;
46
47    let response_body = json!({
48        "status": "ok",
49        "records_processed": result.records_processed,
50        "flush_count": result.batches_flushed,
51        "partitions": result.paths_written,
52    });
53
54    Response::from_json(&response_body)
55}
