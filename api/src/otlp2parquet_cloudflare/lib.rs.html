31/// Durable Object ID separator between signal key and service name.
32pub(crate) const DO_ID_SEPARATOR: char = '|';
33
34/// Version suffix for DO IDs to force fresh instances when needed.
35/// Increment this to invalidate all existing DO instances and create new ones.
36/// v3: Fixed set_alarm to use offset instead of absolute timestamp
37const DO_ID_VERSION: &str = "v3";
38
39/// Create DO ID name from signal key and service name.
40/// Format: "{signal_key}|{service_name}|{version}"
41/// Example: "logs|my-service|v2" or "metrics:gauge|my-service|v2"
42pub(crate) fn make_do_id(signal_key: &SignalKey, service_name: &str) -> String {
43    format!(
44        "{}{}{}{}{}",
45        signal_key, DO_ID_SEPARATOR, service_name, DO_ID_SEPARATOR, DO_ID_VERSION
46    )
47}
48
49/// Parse DO ID into (signal_key_str, service_name).
50/// Returns the raw signal string which can be parsed with SignalKey::from_str().
51pub(crate) fn parse_do_id(id: &str) -> Option<(&str, &str)> {
52    id.split_once(DO_ID_SEPARATOR)
53}

55/// Initialize tracing subscriber for Cloudflare Workers.
56/// Uses tracing-web to output structured JSON logs to the Workers console.
57/// Must be called via #[event(start)] to run once on worker initialization.
58#[event(start)]
59fn init_tracing() {
60    // JSON formatting layer that writes to the Workers console
61    let fmt_layer = tracing_subscriber::fmt::layer()
62        .json()
63        .flatten_event(true) // Put message at top level for $metadata.message in dashboard
64        .with_ansi(false) // ANSI codes not supported in Workers console
65        .with_timer(UtcTime::rfc_3339())
66        .with_writer(MakeConsoleWriter);
67
68    // Performance layer for timing spans (note: spans have identical start/end
69    // times unless they encompass I/O due to Workers platform restrictions)
70    let perf_layer = performance_layer().with_details_from_fields(Pretty::default());
71
72    tracing_subscriber::registry()
73        .with(fmt_layer)
74        .with(perf_layer)
75        .init();
76}

78/// Cloudflare Workers entry point.
79#[event(fetch)]
80pub async fn main(req: Request, env: Env, ctx: Context) -> Result<Response> {
81    tracing::debug!("Worker fetch handler called");
