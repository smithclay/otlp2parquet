42/// Validate catalog configuration with detailed error messages
43fn validate_catalog_config(
44    env: &Env,
45) -> std::result::Result<CatalogMode, errors::ConfigValidationError> {
46    let account_id = env.var("CLOUDFLARE_ACCOUNT_ID").ok().map(|v| v.to_string());
47    let bucket_name = env
48        .var("CLOUDFLARE_BUCKET_NAME")
49        .ok()
50        .map(|v| v.to_string());
51    let api_token = env
52        .secret("CLOUDFLARE_API_TOKEN")
53        .or_else(|_| env.var("CLOUDFLARE_API_TOKEN"))
54        .ok()
55        .map(|v| v.to_string());
56
57    // Collect which variables are present
58    let mut present_vars = Vec::new();
59    if account_id.is_some() {
60        present_vars.push("CLOUDFLARE_ACCOUNT_ID");
61    }
62    if bucket_name.is_some() {
63        present_vars.push("CLOUDFLARE_BUCKET_NAME");
64    }
65    if api_token.is_some() {
66        present_vars.push("CLOUDFLARE_API_TOKEN");
67    }
68
69    match (account_id, bucket_name, api_token) {
70        (Some(a), Some(b), Some(t)) => {
71            // All catalog vars present - use R2 Data Catalog mode
72            Ok(CatalogMode::R2DataCatalog {
73                account_id: a,
74                bucket_name: b,
75                api_token: t,
76            })
77        }
78        (None, None, None) => {
79            // No catalog vars present - use plain Parquet mode
80            Ok(CatalogMode::PlainParquet)
81        }
82        _ => {
83            // Partial configuration - return descriptive error
84            let all_required = vec![
85                "CLOUDFLARE_ACCOUNT_ID",
86                "CLOUDFLARE_BUCKET_NAME",
87                "CLOUDFLARE_API_TOKEN",
88            ];
89            let missing: Vec<&'static str> = all_required
90                .into_iter()
91                .filter(|var| !present_vars.contains(var))
92                .collect();
93
94            Err(errors::ConfigValidationError::PartialCatalogConfig {
95                catalog_type: "R2 Data Catalog",
96                present: present_vars,
97                missing,
98                hint: "R2 Data Catalog requires all three variables to be set. Either provide all catalog variables for Iceberg support, or remove them entirely to use plain Parquet mode.".to_string(),
99            })
100        }
101    }
102}

104/// Decompress gzip-encoded request body if Content-Encoding header indicates gzip
105fn maybe_decompress(data: &[u8], content_encoding: Option<&str>) -> Result<Vec<u8>> {
106    match content_encoding {
107        Some(enc) if enc.eq_ignore_ascii_case("gzip") => {
108            let mut decoder = GzDecoder::new(data);
109            let mut decompressed = Vec::new();
110            decoder.read_to_end(&mut decompressed).map_err(|e| {
111                worker::Error::RustError(format!("gzip decompression failed: {}", e))
112            })?;
113            Ok(decompressed)
114        }
115        _ => Ok(data.to_vec()),
116    }
117}

119/// Cloudflare Workers entry point
120#[event(fetch)]
121pub async fn main(req: Request, env: Env, ctx: Context) -> Result<Response> {
122    console_log!("[DEBUG] Worker fetch handler called");
123    let result = handle_otlp_request(req, env, ctx).await;
124    console_log!("[DEBUG] handle_otlp_request returned");
125    result
126}

128/// Handle OTLP HTTP POST request and write to R2
129async fn handle_otlp_request(mut req: Request, env: Env, _ctx: Context) -> Result<Response> {
130    console_log!("[DEBUG] handle_otlp_request: start");
131    // Use Cloudflare's built-in request ID (CF-Ray header)
132    let request_id = req
133        .headers()
134        .get("cf-ray")
135        .ok()
136        .flatten()
137        .unwrap_or_else(|| "unknown".to_string());
138    console_log!("[DEBUG] request_id: {}", request_id);
139
140    // Only accept POST requests
141    if req.method() != Method::Post {
142        return Response::error("Method not allowed", 405);
143    }
144
145    let path = req.path();
146    console_log!("[DEBUG] path: {}", path);
147    let signal = match path.as_str() {
148        "/v1/logs" => SignalKind::Logs,
149        "/v1/traces" => SignalKind::Traces,
150        "/v1/metrics" => SignalKind::Metrics,
151        _ => return Response::error("Not found", 404),
152    };
153
154    console_log!("[DEBUG] checking auth");
155    // Check basic authentication if enabled
156    if let Err(response) = auth::check_basic_auth(&req, &env, Some(&request_id)) {
157        return Ok(response);
158    }
159
160    console_log!("[DEBUG] loading config");
161    // Load config once (with platform detection and env var overrides)
162    let config = CONFIG
163        .get_or_try_init(|| load_worker_config(&env))
164        .map_err(|e| {
165            console_error!("[{}] Failed to load configuration: {:?}", request_id, e);
166            e
167        })?;
168    console_log!("[DEBUG] config loaded successfully");
169
170    // Initialize storage operator in writer crate if not already done
171    // This is used for direct Parquet writes when no catalog is configured
172    otlp2parquet_writer::initialize_storage(config).map_err(|e| {
173        console_error!("[{}] Failed to initialize storage: {:?}", request_id, e);
174        let error = errors::OtlpErrorKind::StorageError(format!(
175            "Failed to initialize R2 storage: {}. Verify AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY, and OTLP2PARQUET_R2_BUCKET are correctly configured.",
176            e
177        ));
178        let error_response = errors::ErrorResponse::from_error(error, Some(request_id.clone()));
179        worker::Error::RustError(serde_json::to_string(&error_response).unwrap_or_default())
180    })?;
181
182    // Determine whether to use catalog based on catalog_mode configuration
183    // - CatalogMode::Iceberg → use R2 Data Catalog (requires CLOUDFLARE_* env vars)
184    // - CatalogMode::None → write directly to R2 (plain Parquet, no catalog)
185    if config.catalog_mode == otlp2parquet_core::config::CatalogMode::Iceberg
186        && CATALOG.get().is_none()
187    {
188        // Iceberg mode requires CLOUDFLARE_* catalog vars
189        match validate_catalog_config(&env) {
190            Ok(CatalogMode::R2DataCatalog {
191                account_id,
192                bucket_name,
193                api_token,
194            }) => {
195                // All catalog vars present - initialize R2 Data Catalog (Iceberg)
196                let namespace = env
197                    .var("OTLP_NAMESPACE")
198                    .ok()
199                    .map(|v| v.to_string())
200                    .unwrap_or_else(|| "otlp".to_string());
201
202                console_log!(
203                    "[{}] Catalog mode: iceberg - Initializing R2 Data Catalog: account={}, bucket={}, namespace={}",
204                    request_id,
205                    account_id,
206                    bucket_name,
207                    namespace
208                );
209
210                // Note: R2 always uses region="auto" - no need to bridge env vars in WASM
211                // icepick::R2Catalog handles the region internally via OpendDAL configuration
212                // WASM cannot use std::env::set_var anyway (causes panic)
213
214                // Get R2 credentials from config for Parquet file writes
215                let r2_config = config.storage.r2.as_ref().ok_or_else(|| {
216                    console_error!("[{}] R2 storage config missing", request_id);
217                    worker::Error::RustError(
218                        "R2 storage configuration required for catalog mode".to_string(),
219                    )
220                })?;
221
222                let catalog_config = otlp2parquet_writer::CatalogConfig {
223                    namespace: namespace.clone(),
224                    catalog_type: otlp2parquet_writer::CatalogType::R2DataCatalog {
225                        account_id,
226                        bucket_name,
227                        api_token,
228                        access_key_id: r2_config.access_key_id.clone(),
229                        secret_access_key: r2_config.secret_access_key.clone(),
230                    },
231                };
232
233                let catalog = otlp2parquet_writer::initialize_catalog(catalog_config)
234                    .await
235                    .map_err(|e| {
236                        console_error!("[{}] Failed to initialize catalog: {:?}", request_id, e);
237                        let error = errors::OtlpErrorKind::CatalogError(format!(
238                            "Failed to initialize R2 Data Catalog: {}. Verify CLOUDFLARE_ACCOUNT_ID, CLOUDFLARE_BUCKET_NAME, and CLOUDFLARE_API_TOKEN are correct and the API token has R2 permissions.",
239                            e
240                        ));
241                        let error_response = errors::ErrorResponse::from_error(error, Some(request_id.clone()));
242                        worker::Error::RustError(serde_json::to_string(&error_response).unwrap_or_default())
243                    })?;
244
245                // Ensure namespace exists
246                otlp2parquet_writer::ensure_namespace(catalog.as_ref(), &namespace)
247                    .await
248                    .map_err(|e| {
249                        console_error!("[{}] Failed to ensure namespace: {:?}", request_id, e);
250                        let error = errors::OtlpErrorKind::CatalogError(format!(
251                            "Failed to create namespace '{}': {}",
252                            namespace, e
253                        ));
254                        let error_response =
255                            errors::ErrorResponse::from_error(error, Some(request_id.clone()));
256                        worker::Error::RustError(
257                            serde_json::to_string(&error_response).unwrap_or_default(),
258                        )
259                    })?;
260
261                let _ = CATALOG.set(catalog);
262                let _ = NAMESPACE.set(namespace);
263            }
264            Ok(CatalogMode::PlainParquet) => {
265                // Catalog vars not present but iceberg mode requested - this is a config error
266                console_error!(
267                    "[{}] Catalog mode is 'iceberg' but catalog configuration is incomplete",
268                    request_id
269                );
270                let validation_error = errors::ConfigValidationError::MissingRequired {
271                    component: "R2 Data Catalog",
272                    missing_vars: vec!["CLOUDFLARE_ACCOUNT_ID", "CLOUDFLARE_BUCKET_NAME", "CLOUDFLARE_API_TOKEN"],
273                    hint: "Either set all three CLOUDFLARE_* variables for Iceberg mode, or set OTLP2PARQUET_CATALOG_MODE=none to write plain Parquet files".to_string(),
274                };
275                let error = errors::OtlpErrorKind::ConfigError(validation_error);
276                let status_code = error.status_code();
277                let error_response = errors::ErrorResponse::from_error(error, Some(request_id));
278                return error_response.into_response(status_code);
279            }
280            Err(validation_error) => {
281                console_error!(
282                    "[{}] Catalog configuration validation failed: {:?}",
283                    request_id,
284                    validation_error
285                );
286                let error = errors::OtlpErrorKind::ConfigError(validation_error);
287                let status_code = error.status_code();
288                let error_response = errors::ErrorResponse::from_error(error, Some(request_id));
289                return error_response.into_response(status_code);
290            }
291        }
292    } else if config.catalog_mode == otlp2parquet_core::config::CatalogMode::None {
293        // Plain Parquet mode - skip catalog entirely
294        console_log!(
295            "[{}] Catalog mode: none - writing plain Parquet to R2 bucket '{}' (no Iceberg catalog)",
296            request_id,
297            config.storage.r2.as_ref().map(|r2| r2.bucket.as_str()).unwrap_or("unknown")
298        );
299    }
300
301    let catalog = CATALOG.get();
302    let namespace = NAMESPACE.get();
303
304    // Use configurable max payload size (default 10MB for CF Workers)
305    // Can be overridden via OTLP2PARQUET_MAX_PAYLOAD_BYTES env var
306    let max_payload_bytes = env
307        .var("OTLP2PARQUET_MAX_PAYLOAD_BYTES")
308        .ok()
309        .and_then(|val| val.to_string().parse::<usize>().ok())
310        .unwrap_or(config.request.max_payload_bytes);
311
312    console_log!(
313        "Processing {} request (max payload: {} MB)",
314        match signal {
315            SignalKind::Logs => "logs",
316            SignalKind::Traces => "traces",
317            SignalKind::Metrics => "metrics",
318        },
319        max_payload_bytes / (1024 * 1024)
320    );
321
322    let content_type_header = req.headers().get("content-type").ok().flatten();
323    let content_type = content_type_header.as_deref();
324    let format = otlp2parquet_core::InputFormat::from_content_type(content_type);
325    let content_encoding = req.headers().get("content-encoding").ok().flatten();
326
327    let raw_body = req.bytes().await.map_err(|e| {
328        console_error!("Failed to read request body: {:?}", e);
329        e
330    })?;
331
332    // Decompress if gzip-encoded (OTel collectors typically send gzip by default)
333    let body_bytes = maybe_decompress(&raw_body, content_encoding.as_deref())?;
334
335    // Check payload size AFTER decompression (prevents zip bombs)
336    if body_bytes.len() > max_payload_bytes {
337        console_error!(
338            "Payload too large: {} bytes (max: {} bytes / {} MB)",
339            body_bytes.len(),
340            max_payload_bytes,
341            max_payload_bytes / (1024 * 1024)
342        );
343        return Response::error(
344            format!(
345                "Payload too large: {} bytes exceeds limit of {} MB",
346                body_bytes.len(),
347                max_payload_bytes / (1024 * 1024)
348            ),
349            413,
350        );
351    }
352
353    // Determine if catalog is enabled based on config and initialization status
354    let catalog_enabled = config.catalog_mode == otlp2parquet_core::config::CatalogMode::Iceberg;
355
356    // Route to appropriate handler based on signal type
357    match signal {
358        SignalKind::Logs => {
359            handlers::handle_logs_request(
360                &body_bytes,
361                format,
362                content_type,
363                catalog,
364                namespace.map(|s| s.as_str()),
365                catalog_enabled,
366                &request_id,
367            )
368            .await
369        }
370        SignalKind::Traces => {
371            handlers::handle_traces_request(
372                &body_bytes,
373                format,
374                content_type,
375                catalog,
376                namespace.map(|s| s.as_str()),
377                catalog_enabled,
378                &request_id,
379            )
380            .await
381        }
382        SignalKind::Metrics => {
383            handlers::handle_metrics_request(
384                &body_bytes,
385                format,
386                content_type,
387                catalog,
388                namespace.map(|s| s.as_str()),
389                catalog_enabled,
390                &request_id,
391            )
392            .await
393        }
394    }
395}

397fn load_worker_config(env: &Env) -> Result<RuntimeConfig> {
398    console_log!("[DEBUG] load_worker_config: starting");
399    let provider = WorkerEnvSource { env };
400    let inline = provider.get("CONFIG_CONTENT");
401
402    console_log!("[DEBUG] Loading config for Platform::CloudflareWorkers");
403    let config = RuntimeConfig::load_for_platform_with_env(
404        Platform::CloudflareWorkers,
405        inline.as_deref(),
406        &provider,
407    )
408    .map_err(|e| {
409        console_error!("Failed to load configuration: {:?}", e);
410        worker::Error::RustError(format!("Config error: {}", e))
411    })?;
412
413    // Debug: Log loaded configuration
414    console_log!("[DEBUG] Config loaded successfully!");
415    console_log!("[DEBUG] Catalog mode: {:?}", config.catalog_mode);
416    console_log!("[DEBUG] Storage backend: {:?}", config.storage.backend);
417
418    if let Some(ref r2) = config.storage.r2 {
419        console_log!("[DEBUG] R2 Config:");
420        console_log!("[DEBUG]   bucket: {}", r2.bucket);
421        console_log!("[DEBUG]   account_id: {}", r2.account_id);
422        console_log!(
423            "[DEBUG]   access_key_id: {}",
424            if r2.access_key_id.is_empty() {
425                "<empty>"
426            } else {
427                "<set>"
428            }
429        );
430        console_log!(
431            "[DEBUG]   secret_access_key: {}",
432            if r2.secret_access_key.is_empty() {
433                "<empty>"
434            } else {
435                "<set>"
436            }
437        );
438        console_log!("[DEBUG]   endpoint: {:?}", r2.endpoint);
439    } else {
440        console_log!("[DEBUG] R2 Config: None");
441    }
442
443    Ok(config)
444}

446struct WorkerEnvSource<'a> {
447    env: &'a Env,
448}
449
450impl<'a> EnvSource for WorkerEnvSource<'a> {
451    fn get(&self, key: &str) -> Option<String> {
452        let binding = format!("{}{}", ENV_PREFIX, key);
453
454        if let Ok(val) = self.env.var(&binding) {
455            console_log!("[DEBUG] EnvSource: {} = <set>", binding);
456            return Some(val.to_string());
457        }
458        if key.contains("SECRET") {
459            if let Ok(secret) = self.env.secret(&binding) {
460                console_log!("[DEBUG] EnvSource: {} = <secret>", binding);
461                return Some(secret.to_string());
462            }
463        }
464        console_log!("[DEBUG] EnvSource: {} = <not set>", binding);
465        None
466    }
467
468    fn get_raw(&self, key: &str) -> Option<String> {
469        if let Ok(val) = self.env.var(key) {
470            console_log!("[DEBUG] EnvSource (raw): {} = <set>", key);
471            return Some(val.to_string());
472        }
473        if key.contains("SECRET") {
474            if let Ok(secret) = self.env.secret(key) {
475                console_log!("[DEBUG] EnvSource (raw): {} = <secret>", key);
476                return Some(secret.to_string());
477            }
478        }
479        console_log!("[DEBUG] EnvSource (raw): {} = <not set>", key);
480        None
481    }
