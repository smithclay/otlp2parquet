1//! Configuration loading for Cloudflare Workers.
2//!
3//! This module provides the environment variable accessor and storage initialization
4//! logic used by both the main Worker and Durable Objects.
5
6use otlp2parquet_core::config::{EnvSource, Platform, RuntimeConfig, ENV_PREFIX};
7use worker::{Env, Result};
8
9/// Environment source for Cloudflare Workers.
10///
11/// Implements the `EnvSource` trait to provide configuration values from
12/// Cloudflare Workers environment variables and secrets. Used by both
13/// the main Worker fetch handler and Durable Objects.
14pub(crate) struct WorkerEnvSource<'a> {
15    pub env: &'a Env,
16}
17
18/// Check if a key is likely to be a sensitive value that might be stored as a secret.
19fn is_sensitive_key(key: &str) -> bool {
20    key.contains("SECRET")
21        || key.contains("TOKEN")
22        || key.contains("PASSWORD")
23        || key.contains("KEY")
24        || key.contains("CREDENTIAL")
25}
26
27impl EnvSource for WorkerEnvSource<'_> {
28    fn get(&self, key: &str) -> Option<String> {
29        let binding = format!("{}{}", ENV_PREFIX, key);
30
31        // Try var first, then secret for sensitive keys
32        if let Ok(val) = self.env.var(&binding) {
33            tracing::trace!(binding = %binding, "EnvSource: found var");
34            return Some(val.to_string());
35        }
36        if is_sensitive_key(key) {
37            if let Ok(secret) = self.env.secret(&binding) {
38                tracing::trace!(binding = %binding, "EnvSource: found secret");
39                return Some(secret.to_string());
40            }
41        }
42        tracing::trace!(binding = %binding, "EnvSource: not set");
43        None
44    }
45
46    fn get_raw(&self, key: &str) -> Option<String> {
47        // Get variable without prefix (for AWS_* vars, etc.)
48        match self.env.var(key) {
49            Ok(val) => {
50                let value = val.to_string();
51                tracing::info!(key = %key, len = value.len(), "EnvSource (raw): found var");
52                return Some(value);
53            }
54            Err(e) => {
55                tracing::debug!(key = %key, error = %e, "EnvSource (raw): var not found, checking secret");
56            }
57        }
58        // Check secrets for sensitive keys (TOKEN, SECRET, KEY, PASSWORD, CREDENTIAL)
59        let is_sensitive = is_sensitive_key(key);
60        tracing::info!(key = %key, is_sensitive = %is_sensitive, "EnvSource (raw): checking if sensitive");
61        if is_sensitive {
62            match self.env.secret(key) {
63                Ok(secret) => {
64                    let value = secret.to_string();
65                    tracing::info!(key = %key, len = value.len(), "EnvSource (raw): found secret");
66                    return Some(value);
67                }
68                Err(e) => {
69                    tracing::error!(key = %key, error = %e, "EnvSource (raw): secret lookup FAILED");
70                }
71            }
72        }
73        tracing::warn!(key = %key, "EnvSource (raw): not found anywhere");
74        None
75    }
76}

78/// Load config and initialize storage for the Durable Object context.
79///
80/// This must be called before any write operations since DOs run in their own isolate.
81/// Returns the RuntimeConfig for use in platform-specific configuration needs.
82///
83/// # Errors
84///
85/// Returns an error if configuration loading or storage initialization fails.
86pub(crate) fn ensure_storage_initialized(env: &Env) -> Result<RuntimeConfig> {
87    let provider = WorkerEnvSource { env };
88    let inline = provider.get("CONFIG_CONTENT");
89
90    let config = RuntimeConfig::load_for_platform_with_env(
91        Platform::CloudflareWorkers,
92        inline.as_deref(),
93        &provider,
94    )
95    .map_err(|e| worker::Error::RustError(format!("DO config error: {}", e)))?;
96
97    tracing::info!("Config loaded in DO context");
98
99    otlp2parquet_writer::initialize_storage(&config)
100        .map_err(|e| worker::Error::RustError(format!("DO storage init error: {}", e)))?;
101
102    Ok(config)
103}
