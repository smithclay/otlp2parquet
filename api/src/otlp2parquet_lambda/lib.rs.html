3// Uses OpenDAL S3 for storage and handles Lambda function events

22/// Lambda handler for OTLP HTTP requests
23async fn handle_request(
24    event: LambdaEvent<HttpRequestEvent>,
25    state: Arc<LambdaState>,
26) -> Result<HttpLambdaResponse, Error> {
27    let (request, _context) = event.into_parts();
28
29    match request {
30        HttpRequestEvent::ApiGateway(boxed_request) => {
31            let request = &*boxed_request;
32            let method = request.http_method.as_str();
33            let path = canonical_path(request.path.as_deref());
34
35            // Extract Content-Type header from API Gateway request
36            let content_type = request
37                .headers
38                .get("content-type")
39                .and_then(|v| v.to_str().ok());
40
41            let response = handle_http_request(
42                method,
43                &path,
44                request.body.as_deref(),
45                request.is_base64_encoded,
46                content_type,
47                &state,
48            )
49            .await;
50            Ok(build_api_gateway_response(response))
51        }
52        HttpRequestEvent::FunctionUrl(boxed_request) => {
53            let request = &*boxed_request;
54            let method = request
55                .request_context
56                .http
57                .method
58                .as_deref()
59                .unwrap_or("GET");
60            let path = canonical_path(
61                request
62                    .raw_path
63                    .as_deref()
64                    .or(request.request_context.http.path.as_deref()),
65            );
66
67            // Extract Content-Type header from Function URL request
68            let content_type = request
69                .headers
70                .get("content-type")
71                .and_then(|v| v.to_str().ok());
72
73            let response = handle_http_request(
74                method,
75                &path,
76                request.body.as_deref(),
77                request.is_base64_encoded,
78                content_type,
79                &state,
80            )
81            .await;
82            Ok(build_function_url_response(response))
83        }
84    }
85}

87#[derive(Clone)]
88pub(crate) struct LambdaState {
89    pub parquet_writer: Arc<ParquetWriter>,
90    pub passthrough: PassthroughBatcher,
91    pub max_payload_bytes: usize,
92}

94/// Lambda runtime entry point
95pub async fn run() -> Result<(), Error> {
96    println!("Lambda runtime - using lambda_runtime's tokio + OpenDAL S3");
97
98    // Load configuration
99    let config = RuntimeConfig::load()
100        .map_err(|e| lambda_runtime::Error::from(format!("Failed to load configuration: {}", e)))?;
101
102    // Validate storage backend is S3
103    if config.storage.backend != StorageBackend::S3 {
104        return Err(lambda_runtime::Error::from(format!(
105            "Lambda requires S3 storage backend, got: {}",
106            config.storage.backend
107        )));
108    }
109
110    let s3 = config
111        .storage
112        .s3
113        .as_ref()
114        .ok_or_else(|| lambda_runtime::Error::from("S3 configuration required for Lambda"))?;
115
116    set_parquet_row_group_size(config.storage.parquet_row_group_size);
117
118    // Initialize OpenDAL S3 storage
119    // OpenDAL automatically discovers AWS credentials from:
120    // - IAM role (preferred for Lambda)
121    // - Environment variables (AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY)
122    // - AWS credentials file
123    let storage = Arc::new(
124        otlp2parquet_storage::opendal_storage::OpenDalStorage::new_s3(
125            &s3.bucket,
126            &s3.region,
127            s3.endpoint.as_deref(),
128            None,
129            None,
130        )
131        .map_err(|e| lambda_runtime::Error::from(format!("Failed to initialize storage: {}", e)))?,
132    );
133    let parquet_writer = Arc::new(ParquetWriter::new(storage.operator().clone()));
134
135    // Lambda: Always use passthrough (no batching)
136    // Event-driven/stateless architecture makes batching ineffective
137    println!("Lambda using passthrough mode (no batching)");
138
139    let max_payload_bytes = config.request.max_payload_bytes;
140    println!("Lambda payload cap set to {} bytes", max_payload_bytes);
141
142    let state = Arc::new(LambdaState {
143        parquet_writer,
144        passthrough: PassthroughBatcher::default(),
145        max_payload_bytes,
146    });
147
148    lambda_runtime::run(service_fn(move |event: LambdaEvent<HttpRequestEvent>| {
149        let state = state.clone();
150        async move { handle_request(event, state).await }
151    }))
152    .await
153}
